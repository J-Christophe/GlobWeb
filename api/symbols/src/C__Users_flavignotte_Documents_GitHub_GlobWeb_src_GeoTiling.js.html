 22 /** @constructor
 23 	GeoTiling constructor
 24  */
 25 GlobWeb.GeoTiling = function(nx,ny)
 26 {
 27 	this.level0NumTilesX = nx;
 28 	this.level0NumTilesY = ny;
 29 }

 33 /** 
 34 	Generate the tiles for level zero
 35  */
 36 GlobWeb.GeoTiling.prototype.generateLevelZeroTiles = function(config)
 37 {	
 38 	config.skirt = 1;
 39 	config.cullSign = 1;
 40 
 41 	var level0Tiles = [];
 42 	
 43 	var latStep = 180 / this.level0NumTilesY;
 44 	var lonStep = 360 / this.level0NumTilesX;
 45 	
 46 	for (var j = 0; j < this.level0NumTilesY; j++)
 47 	{
 48 		for (var i = 0; i < this.level0NumTilesX; i++)
 49 		{
 50 			var geoBound = new GlobWeb.GeoBound( -180 + i * lonStep, 90 - (j+1) * latStep, -180 + (i+1) * lonStep, 90 - j * latStep  );
 51 			var tile = new GlobWeb.GeoTile(geoBound, 0, i, j)
 52 			tile.config = config;
 53 			level0Tiles.push( tile );
 54 		}
 55 	}
 56 
 57 	return level0Tiles;
 58 }

 62 /** 
 63 	Locate a level zero tile
 64  */
 65 GlobWeb.GeoTiling.prototype.lonlat2LevelZeroIndex = function(lon,lat)
 66 {	
 67 	var i = Math.floor( (lon + 180) * this.level0NumTilesX / 360 );
 68  	var j = Math.floor( (90 - lat) * this.level0NumTilesY / 180 );
 69 	return j * this.level0NumTilesX + i;
 70 
 71 }

 78 GlobWeb.GeoTile = function( geoBound, level, x, y )

 96 /** @export
 97   Get elevation at a geo position
 98 */
 99 GlobWeb.GeoTile.prototype.getElevation = function(lon,lat)
100 {
101 	// Get the lon/lat in coordinates between [0,1] in the tile
102 	var u = (lon - this.geoBound.west) / (this.geoBound.east - this.geoBound.west);
103  	var v = (lat - this.geoBound.north) / (this.geoBound.south - this.geoBound.north);
104 
105 	// Quick fix when lat is on the border of the tile
106 	var childIndex = (v >= 1 ? 1 : Math.floor(2*v) )*2 + Math.floor(2*u);
107 	if ( this.children && this.children[childIndex].state == GlobWeb.Tile.State.LOADED )
108 		return this.children[childIndex].getElevation(lon,lat);
109 	
110 	var tess = this.config.tesselation;
111 	var i = Math.floor( u * tess );
112 	var j = Math.floor( v * tess );
113 	
114 	var vo = this.config.vertexSize * (j * tess + i);
115 	var vertex = [ this.vertices[vo], this.vertices[vo+1], this.vertices[vo+2] ];
116 	mat4.multiplyVec3( this.matrix, vertex );
117 	var geo = GlobWeb.CoordinateSystem.from3DToGeo(vertex);
118 	return geo[2];
119 }

123 /**
124 	Create the children
125  */
126 GlobWeb.GeoTile.prototype.createChildren = function()
127 {
128 	// Create the children
129 	var lonCenter = ( this.geoBound.east + this.geoBound.west ) * 0.5;
130 	var latCenter = ( this.geoBound.north + this.geoBound.south ) * 0.5;
131 	
132 	var level = this.level+1;
133 	
134 	var tile00 = new GlobWeb.GeoTile( new GlobWeb.GeoBound( this.geoBound.west, latCenter, lonCenter, this.geoBound.north), level, 2*this.x, 2*this.y );
135 	var tile10 = new GlobWeb.GeoTile( new GlobWeb.GeoBound( lonCenter, latCenter,  this.geoBound.east, this.geoBound.north), level, 2*this.x+1, 2*this.y );
136 	var tile01 = new GlobWeb.GeoTile( new GlobWeb.GeoBound( this.geoBound.west, this.geoBound.south, lonCenter, latCenter), level, 2*this.x, 2*this.y+1 );
137 	var tile11 = new GlobWeb.GeoTile( new GlobWeb.GeoBound( lonCenter, this.geoBound.south, this.geoBound.east, latCenter), level, 2*this.x+1, 2*this.y+1  );
138 	
139 	tile00.initFromParent( this, 0, 0 );
140 	tile10.initFromParent( this, 1, 0 );
141 	tile01.initFromParent( this, 0, 1 );
142 	tile11.initFromParent( this, 1, 1 );
143 	
144 	this.children = [ tile00, tile10, tile01, tile11 ];	
145 }

149 /**
150 	Convert coordinates in longitude,latitude to coordinate in "tile space"
151 	Tile space means coordinates are between [0,tesselation-1] if inside the tile
152 	Used by renderers algorithm to clamp coordinates on the tile
153  */
154 GlobWeb.GeoTile.prototype.lonlat2tile = function(coordinates)
155 {
156 	var ul = this.geoBound.east - this.geoBound.west;
157 	var vl = this.geoBound.south - this.geoBound.north;
158 	var factor = this.config.tesselation-1;
159 	
160 	var tileCoords = [];
161 	for ( var i = 0; i < coordinates.length; i++ )
162 	{
163 		var u = factor * (coordinates[i][0] - this.geoBound.west) / ul;
164 		var v = factor * (coordinates[i][1] - this.geoBound.north) / vl;
165 		tileCoords.push( [ u, v ] );	
166 	}
167 	
168 	return tileCoords;
169 }

173 /**
174 	Generate vertices for tile
175  */
176 GlobWeb.GeoTile.prototype.generateVertices = function(elevations)
177 {	
178 	// Compute tile matrix
179 	this.matrix = GlobWeb.CoordinateSystem.getLHVTransform( this.geoBound.getCenter() );
180 	var invMatrix = mat4.create();
181 	mat4.inverse( this.matrix, invMatrix );
182 	this.inverseMatrix = invMatrix;
183 	
184 	// Build the vertices
185 	var vertexSize = this.config.vertexSize;
186 	var size = this.config.tesselation;
187 	var vertices = new Float32Array( vertexSize*size*(size+6) );
188 	var lonStep = (this.geoBound.east - this.geoBound.west) / (size-1);
189 	var latStep = (this.geoBound.south - this.geoBound.north) / (size-1);
190 	var radius = GlobWeb.CoordinateSystem.radius;
191 	var scale = GlobWeb.CoordinateSystem.heightScale;
192 	var offset = 0;
193 	
194 	var lat = this.geoBound.north * Math.PI / 180.0;
195 	latStep = latStep * Math.PI / 180.0;
196 	lonStep = lonStep * Math.PI / 180.0;
197 	for ( var j=0; j < size; j++)
198 	{
199 		var cosLat = Math.cos( lat );
200 		var sinLat = Math.sin( lat );
201 		
202 		var lon = this.geoBound.west * Math.PI / 180.0;
203 				
204 		for ( var i=0; i < size; i++)
205 		{
206 			var height = elevations ? scale * elevations[ offset ] : 0.0;
207 			
208 			var x = (radius + height) * Math.cos( lon ) * cosLat;
209 			var y = (radius + height) * Math.sin( lon ) * cosLat;
210 			var z = (radius + height) * sinLat;
211 			
212 			var vi = offset * vertexSize;
213 			vertices[vi] = invMatrix[0]*x + invMatrix[4]*y + invMatrix[8]*z + invMatrix[12];
214 			vertices[vi+1] = invMatrix[1]*x + invMatrix[5]*y + invMatrix[9]*z + invMatrix[13];
215 			vertices[vi+2] = invMatrix[2]*x + invMatrix[6]*y + invMatrix[10]*z + invMatrix[14];
216 						
217 			offset++;
218 			lon += lonStep;
219 		}
220 		
221 		lat += latStep;
222 	}
223 	
224 	return vertices;
225 }
