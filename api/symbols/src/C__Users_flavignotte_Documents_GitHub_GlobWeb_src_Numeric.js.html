 30 /**
 31   Linear interpolation between [a, b], t must be [0, 1]
 32 */
 33 Numeric.lerp = function(t, a, b)
 34 {
 35     return a + ((b - a) * t);
 36 }

 40 /**
 41  Cubic interpolation between [a, b], t must be [0, 1]
 42 */
 43 Numeric.cubicInterpolation = function(t, startPos, startVel, endPos, endVel)
 44 {
 45 	var t2 = t * t;
 46 	var t3 = t2 * t; 
 47 	
 48 	// Evaluate the position
 49 	
 50 	var M00 = 2 * startPos[0] - 2 * endPos[0] + startVel[0] + endVel[0];
 51 	var M10 = 2 * startPos[1] - 2 * endPos[1] + startVel[1] + endVel[1];
 52 	var M20 = 2 * startPos[2] - 2 * endPos[2] + startVel[2] + endVel[2];
 53 
 54 	var M01 = -3 * startPos[0] + 3 * endPos[0] - 2 * startVel[0] - endVel[0];
 55 	var M11 = -3 * startPos[1] + 3 * endPos[1] - 2 * startVel[1] - endVel[1];
 56 	var M21 = -3 * startPos[2] + 3 * endPos[2] - 2 * startVel[2] - endVel[2];
 57 	
 58 	var position = vec3.create();
 59 	position[0] = M00 * t3 + M01 * t2 + startVel[0] * t + startPos[0];
 60 	position[1] = M10 * t3 + M11 * t2 + startVel[1] * t + startPos[1];
 61 	position[2] = M20 * t3 + M21 * t2 + startVel[2] * t + startPos[2];
 62 	
 63 	return position;
 64 }

 68 /**
 69  Cubic interpolation between [a, b], t must be [0, 1]
 70 */
 71 Numeric.cubicInterpolationDerivative = function(t, startPos, startVel, endPos, endVel)
 72 {
 73 	var t2 = t * t;
 74 	
 75 	// Evaluates the direction
 76 
 77 	var M01 = 6 * startPos[0] - 6 * endPos[0] + 3 * startVel[0] + 3 * endVel[0];
 78 	var M11 = 6 * startPos[1] - 6 * endPos[1] + 3 * startVel[1] + 3 * endVel[1];
 79 	var M21 = 6 * startPos[2] - 6 * endPos[2] + 3 * startVel[2] + 3 * endVel[2];
 80 
 81 	var M02 = -6 * startPos[0] + 6 * endPos[0] - 4 * startVel[0] - 2 * endVel[0];
 82 	var M12 = -6 * startPos[1] + 6 * endPos[1] - 4 * startVel[1] - 2 * endVel[1];
 83 	var M22 = -6 * startPos[2] + 6 * endPos[2] - 4 * startVel[2] - 2 * endVel[2];
 84 
 85 	var direction = vec3.create();
 86 	direction[0] = M01 * t2 + M02 * t + startVel[0];
 87 	direction[1] = M11 * t2 + M12 * t + startVel[1];
 88 	direction[2] = M21 * t2 + M22 * t + startVel[2];
 89 	
 90 	return direction;
 91 }

 95 /**
 96   Map x between [xMin, xMax] to [0, 1]
 97 */
 98 Numeric.map01 = function(x, xMin, xMax)
 99 {
100     return (xMin != xMax) ? (x - xMin) / (xMax - xMin) : 0;
101 }

105 /*
106   Map x between [xMin, xMax] to [outMin, outMax]
107 */
108 Numeric.mapLinear = function(x, xMin, xMax, outMin, outMax)
109 {
110     return Numeric.lerp(Numeric.map01(x, xMin, xMax), outMin, outMax);
111 }

131 /**
132   Remap input t ([0, 1]) to a curve starting slowly
133   and accelerating till 0.5 an decelerating till 1
134 */
135 Numeric.easeInOutQuad = function(t)
136 {
137     var out = t;
138     if (out < 0.5)
139     {
140         // use (0.5*t^2) with input [0, 1]
141         out = out+out; // map [0 0.5] outo [0 1]
142         out = 0.5*(out*out);
143     }
144     else
145     {
146         // use (0.5*(1-t)^2) with input [-1, 0]
147         out = (out+out) - 2.0; // map [0.5 1] to [-1 0]
148         out = 0.5*(1.0-(out*out));
149         out = 0.5 + out;
150     }
151     return out;
152 }

157  */
158 Numeric.easeOutInQuad = function(t)
159 {
160     var out = t;
161     if (out < 0.5)
162     {
163         // use (0.5*(1-t)^2) with input [-1, 0]
164         out = (out+out) - 1.0; // map [0 0.5] to [-1 0]
165         out = 0.5*(1.0-(out*out));
166     }
167     else
168     {
169         // use (0.5*t^2) with input [0, 1]
170         out = (out+out) - 1.0; // map [0.5 1] outo [0 1]
171         out = 0.5*(out*out);
172         out = 0.5 + out;
173     }
174     return out;
175 }

199 /**
200   Computes point on a ray
201 */
202 Numeric.pointOnRay = function(rayOrigin, rayDirection, t, dest)
203 {
204     if (!dest) { dest = vec3.create(); }
205 
206     vec3.scale(rayDirection, t, dest);
207     vec3.add(dest, rayOrigin, dest);
208 
209     return dest;
210 }

214 /**
215   Line-line intersection
216   rayDirection must be normalized.
217   Returns t at which intersection occurs or -1 if no intersection.
218 */
219 
220 Numeric.lineIntersection = function( x1, y1, x2, y2, x3, y3, x4, y4 )
221 {
222 	var det = (y4 - y3) * (x2 - x1) - (x4 - x3) * (y2 - y1);
223 	if ( det == 0 )
224 	{
225 		return [-1,-1];
226 	}
227 	
228 	var ua = (x4 - x3) * (y1 - y3) - (y4 - y3) * (x1 - x3);
229 	var ub = (x2 - x1) * (y1 - y3) - (y2 - y1) * (x1 - x3);
230 	
231 	ua /= det;
232 	ub /= det;
233 	
234 	return [ ua, ub ];
235 	//return ua > 0.0 && ua < 1.0 && ub > 0.0 && ub < 1.0;
236 }

240 /**
241   Ray sphere intersection
242   rayDirection must be normalized.
243   Returns t at which intersection occurs or -1 if no intersection.
244 */
245 Numeric.raySphereIntersection = function(rayOrigin, rayDirection, sphereCenter, sphereRadius)
246 {
247     // cf. http://wiki.cgsociety.org/index.php/Ray_Sphere_Intersection
248 
249     var rs = vec3.subtract( rayOrigin, sphereCenter, vec3.create() );
250     // rayDirection is normalized so a = 1
251     // var a = vec3.dot(rayDirection, rayDirection);
252     var b = 2.0 * vec3.dot(rayDirection, rs);
253     var c = vec3.dot(rs, rs) - sphereRadius*sphereRadius;
254 
255     // as a == 1, discriminant = b^2 - (4*c)
256     // var discr = (b*b) - (4*a*c);
257     var discr = (b*b) - (4*c);
258     if (discr < 0)
259         return -1;
260 
261     // t0 = (-b - sqrt(discr)) / 2a, t1 = (-b + sqrt(discr)) / 2a, a == 1
262     discr = Math.sqrt(discr);
263     var tNear = (-b - discr) / 2;
264     var tFar  = (-b + discr) / 2;
265     if (tNear > tFar) // Swap t values
266     {
267         var tmp = tNear;
268         tNear = tFar;
269         tFar = tmp;
270     }
271 
272     if (tFar < 0) // Hit is beyond ray origin
273         return -1;
274     
275     return tNear < 0 ? tFar : tNear;
276 }

298 /**
299   Project a vec3
300 */
301 mat4.project = function(mat, vec, dest)
302 {
303  	if(!dest) { dest = vec }
304 	mat4.multiplyVec4( mat, vec, dest );
305 	var iw = 1.0 / dest[3];
306 	dest[0] *= iw;
307 	dest[1] *= iw;
308 	dest[2] *= iw;
309 	return dest;
310 }

312 /**
313  * mat4.rotateVec3
314  * Rotate a vec3 with the given matrix
315  *
316  * Params:
317  * mat - mat4 to transform the vector with
318  * vec - vec3 to transform
319  * dest - Optional, vec3 receiving operation result. If not specified result is written to vec
320  *
321  * Returns:
322  * dest if specified, vec otherwise
323  */
324 mat4.rotateVec3 = function(mat, vec, dest) {
325 	if(!dest) { dest = vec }
326 	
327 	var x = vec[0], y = vec[1], z = vec[2];
328 	
329 	dest[0] = mat[0]*x + mat[4]*y + mat[8]*z;
330 	dest[1] = mat[1]*x + mat[5]*y + mat[9]*z;
331 	dest[2] = mat[2]*x + mat[6]*y + mat[10]*z;
332 	
333 	return dest;
334 };

338 /**
339   Update a matrix with the given position, attitude parameters
340 */
341 mat4.fromPositionAttitude = function(position, attitude, dest)
342 {
343     if (!dest) { dest = mat4.create(); }
344 
345     // This sets the translation part to 0
346     quat4.trueToMat4(attitude, dest);
347     // Update translation
348     dest[12] = position[0];
349     dest[13] = position[1];
350     dest[14] = position[2];
351 
352     return dest;
353 }

357 /**
358   Create a rotation quaternion from the given rotation matrix
359 */
360 mat3.toQuat4 = function(mat, dest)
361 {
362     if (!dest) { dest = quat4.create(); }
363 
364     var trace = mat[0] + mat[4] + mat[8];
365     if (trace > 0) { 
366         var S = Math.sqrt(trace + 1.0) * 2; // S=4*qw 
367         dest[3] = 0.25 * S;
368         dest[0] = (mat[5] - mat[7]) / S;
369         dest[1] = (mat[6] - mat[2]) / S;
370         dest[2] = (mat[1] - mat[3]) / S;
371     } else if ((mat[0] > mat[4]) && (mat[0] > mat[8])) { 
372         var S = Math.sqrt(1.0 + mat[0] - mat[4] - mat[8]) * 2; // S=4*qx 
373         dest[3] = (mat[5] - mat[7]) / S;
374         dest[0] = 0.25 * S;
375         dest[1] = (mat[3] + mat[1]) / S; 
376         dest[2] = (mat[6] + mat[2]) / S; 
377     } else if (mat[4] > mat[8]) { 
378         var S = Math.sqrt(1.0 + mat[4] - mat[0] - mat[8]) * 2; // S=4*qy
379         dest[3] = (mat[6] - mat[2]) / S;
380         dest[0] = (mat[3] + mat[1]) / S; 
381         dest[1] = 0.25 * S;
382         dest[2] = (mat[7] + mat[5]) / S; 
383     } else { 
384         var S = Math.sqrt(1.0 + mat[8] - mat[0] - mat[4]) * 2; // S=4*qz
385         dest[3] = (mat[1] - mat[3]) / S;
386         dest[0] = (mat[6] + mat[2]) / S;
387         dest[1] = (mat[7] + mat[5]) / S;
388         dest[2] = 0.25 * S;
389     }
390 
391     return dest;
392 }

435 /**
436   Fixed quaternion to rotation matrix function
437 */
438 quat4.trueToMat4 = function(quat, dest) {
439         if(!dest) { dest = mat4.create(); }
440         
441         var x = quat[0], y = quat[1], z = quat[2], w = quat[3];
442 
443         var x2 = x + x;
444         var y2 = y + y;
445         var z2 = z + z;
446 
447         var xx = x*x2;
448         var xy = x*y2;
449         var xz = x*z2;
450 
451         var yy = y*y2;
452         var yz = y*z2;
453         var zz = z*z2;
454 
455         var wx = w*x2;
456         var wy = w*y2;
457         var wz = w*z2;
458 
459         dest[0] = 1 - (yy + zz);
460         dest[1] = xy + wz;
461         dest[2] = xz - wy;
462         dest[3] = 0;
463 
464         dest[4] = xy - wz;
465         dest[5] = 1 - (xx + zz);
466         dest[6] = yz + wx;
467         dest[7] = 0;
468 
469         dest[8] = xz + wy;
470         dest[9] = yz - wx;
471         dest[10] = 1 - (xx + yy);
472         dest[11] = 0;
473 
474         dest[12] = 0;
475         dest[13] = 0;
476         dest[14] = 0;
477         dest[15] = 1;
478         
479         return dest;
480 }

484 /**
485   Fixed quaternion to rotation matrix function
486 */
487 quat4.trueToMat3 = function(quat, dest) {
488         if(!dest) { dest = mat3.create(); }
489         
490         var x = quat[0], y = quat[1], z = quat[2], w = quat[3];
491 
492         var x2 = x + x;
493         var y2 = y + y;
494         var z2 = z + z;
495 
496         var xx = x*x2;
497         var xy = x*y2;
498         var xz = x*z2;
499 
500         var yy = y*y2;
501         var yz = y*z2;
502         var zz = z*z2;
503 
504         var wx = w*x2;
505         var wy = w*y2;
506         var wz = w*z2;
507 
508         dest[0] = 1 - (yy + zz);
509         dest[1] = xy + wz;
510         dest[2] = xz - wy;
511 
512         dest[3] = xy - wz;
513         dest[4] = 1 - (xx + zz);
514         dest[5] = yz + wx;
515 
516         dest[6] = xz + wy;
517         dest[7] = yz - wx;
518         dest[8] = 1 - (xx + yy);
519         
520         return dest;
521 }

525 /**
526   Fixed slerp quaternion function (taken from Ogre3D sources)
527 */
528 quat4.trueSlerp = function(q1, q2, t, dest)
529 {
530     var cosAngle = q1[0]*q2[0] + q1[1]*q2[1] + q1[2]*q2[2] + q1[3]*q2[3];
531     if (!dest) { dest = quat4.create(); }
532 
533     // Do we need to invert rotation to use the shortest path ?
534     if (cosAngle < 0.0)
535     {
536         cosAngle = -cosAngle;
537         dest[0] = -q2[0];
538         dest[1] = -q2[1];
539         dest[2] = -q2[2];
540         dest[3] = -q2[3];
541     }
542     else
543     {
544         dest[0] = q2[0];
545         dest[1] = q2[1];
546         dest[2] = q2[2];
547         dest[3] = q2[3];
548     }
549 
550     if (Math.abs(cosAngle) < (1 - 0.0001)) // epsilon
551     {
552         // Standard case (slerp)
553         var sinAngle = Math.sqrt(1 - cosAngle*cosAngle);
554         var angle = Math.atan2(sinAngle, cosAngle);
555         var c0 = Math.sin((1 - t) * angle) / sinAngle;
556         var c1 = Math.sin(t * angle) / sinAngle;
557 
558         dest[0] = q1[0]*c0 + c1*dest[0];
559         dest[1] = q1[1]*c0 + c1*dest[1];
560         dest[2] = q1[2]*c0 + c1*dest[2];
561         dest[3] = q1[3]*c0 + c1*dest[3];
562     }
563     else
564     {
565         // There are two situations:
566         // 1. "q1" and "q2" are very close (cosAngle ~= +1), so we can do a linear
567         //    interpolation safely.
568         // 2. "q1" and "q2" are almost inverse of each other (cosAngle ~= -1), there
569         //    are an infinite number of possibilities interpolation. but we haven't
570         //    have method to fix this case, so just use linear interpolation here.
571         dest[0] = Numeric.lerp(t, q1[0], q2[0]);
572         dest[1] = Numeric.lerp(t, q1[1], q2[1]);
573         dest[2] = Numeric.lerp(t, q1[2], q2[2]);
574         dest[3] = Numeric.lerp(t, q1[3], q2[3]);
575         // taking the complement requires renormalisation
576         quat4.normalize(dest);
577     }    
578 
579     return dest;
580 }
