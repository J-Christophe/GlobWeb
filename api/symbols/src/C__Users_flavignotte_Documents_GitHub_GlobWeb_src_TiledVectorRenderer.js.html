 23 /** @constructor
 24 	TiledVectorRenderer constructor
 25  */
 26 GlobWeb.TiledVectorRenderer = function(tileManager)
 27 {
 28 	this.tileManager = tileManager;
 29 	
 30 	// Create a bucket with default style
 31 	// Bucket aggregate geometries that shares a common style
 32 	this.buckets = [  { style: new GlobWeb.FeatureStyle(), geometries: [] } ];
 33 	
 34 	var vertexShader = "\
 35 	attribute vec3 vertex; \n\
 36 	uniform float zOffset; \n\
 37 	uniform mat4 modelViewMatrix;\n\
 38 	uniform mat4 projectionMatrix;\n\
 39 	\n\
 40 	void main(void)  \n\
 41 	{ \n\
 42 		gl_Position = projectionMatrix * modelViewMatrix * vec4(vertex.x, vertex.y, vertex.z + zOffset, 1.0); \n\
 43 	} \n\
 44 	";
 45 	
 46 	var fragmentShader = "\
 47 	#ifdef GL_ES \n\
 48 	precision highp float; \n\
 49 	#endif \n\
 50 	uniform vec4 color; \n\
 51 	\n\
 52 	void main(void) \n\
 53 	{ \n\
 54 		gl_FragColor = color; \n\
 55 	} \n\
 56 	";
 57 
 58     this.program = new GlobWeb.Program(this.tileManager.renderContext);
 59     this.program.createFromSource(vertexShader, fragmentShader);
 60 	
 61 	// Customization for different renderer : lineString or polygon
 62 	this.styleEquals = null;
 63 	this.renderableConstuctor = null;
 64 	this.id = "empty";
 65 }

 69 /**
 70 	Get or create a bucket to store a feature with the given style
 71  */
 72 GlobWeb.TiledVectorRenderer.prototype.getOrCreateBucket = function( layer, style )
 73 {
 74 	for ( var i = 0; i < this.buckets.length; i++ )
 75 	{
 76 		if ( this.buckets[i].layer == layer && this.styleEquals(style, this.buckets[i].style ) )
 77 		{
 78 			return this.buckets[i];
 79 		}
 80 	}
 81 	
 82 	var bucket = { layer: layer, style: style, geometries: [] };
 83 	this.buckets.push( bucket );
 84 	return bucket;
 85 }

 89 /**
 90 	Get or create a renderable for a tile
 91  */
 92 GlobWeb.TiledVectorRenderer.prototype.findRenderable = function( bucket, tile )
 93 {
 94 	if ( tile.extension[this.id] )
 95 	{
 96 		var renderables = tile.extension[this.id].renderables;
 97 		for ( var i = 0; i < renderables.length; i++ )
 98 		{
 99 			if ( renderables[i].bucket == bucket )
100 			{
101 				return renderables[i];
102 			}
103 		}
104 	}
105 
106 	return null;
107 }

111 /**
112 	Remove a geometry from the tile
113  */
114 GlobWeb.TiledVectorRenderer.prototype.removeGeometryFromTile = function( bucket, geometry, tile )
115 {
116 	var renderable = this.findRenderable( bucket, tile );
117 	if ( renderable && renderable.removeGeometry( geometry ) && tile.children )
118 	{
119 		// Remove the geometry from loaded children
120 		for ( var i = 0; i < 4; i++ )
121 		{
122 			if ( tile.children[i].state == GlobWeb.Tile.State.LOADED )
123 			{
124 				this.removeGeometryFromTile( bucket, geometry, tile.children[i] );
125 			}
126 		}
127 	}
128 }

132 /**
133 	Remove a geometry from the renderer
134  */
135 GlobWeb.TiledVectorRenderer.prototype.removeGeometry = function( geometry, layer )
136 {
137 	var foundBucket = null;
138 	
139 	// Remove geometry from buckets
140 	for ( var i = 0; i < this.buckets.length && !foundBucket; i++ )
141 	{
142 		var bucket = this.buckets[i];
143 		if ( bucket.layer == layer ) 
144 		{
145 			var index = bucket.geometries.indexOf( geometry );
146 			if ( index != -1 )
147 			{
148 				bucket = this.buckets[i];
149 				bucket.geometries.splice( index, 1 );
150 				foundBucket = bucket;
151 			}
152 		}
153 	}
154 	
155 	// Remove geometry 
156 	if ( foundBucket )
157 	{
158 		for ( var i = 0; i < this.tileManager.level0Tiles.length; i++ )
159 		{
160 			this.removeGeometryFromTile( foundBucket, geometry, this.tileManager.level0Tiles[i] );
161 		}
162 	}
163 }

182 /**
183 	Add a geometry to the renderer.
184 	Public method to add geometry to the renderer
185  */
186 GlobWeb.TiledVectorRenderer.prototype.addGeometry = function( geometry, layer, style )
187 {
188 	var bucket = this.getOrCreateBucket( layer, style );
189 	bucket.geometries.push( geometry );
190 	
191 	for ( var i = 0; i < this.tileManager.level0Tiles.length; i++ )
192 	{
193 		var tile = this.tileManager.level0Tiles[i];
194 		if ( tile.state == GlobWeb.Tile.State.LOADED )
195 			this.addGeometryToTile( bucket, geometry, tile );
196 	}
197 }

201 /**
202 	Add a geometry to the given tile.
203 	The method is recursive, it will also add the geometry to children if exists
204  */
205 GlobWeb.TiledVectorRenderer.prototype.addGeometryToTile = function( bucket, geometry, tile )
206 {
207 	var isNewRenderable = false;
208 	
209 	// Try to find an existing renderable on the tile
210 	var renderable = this.findRenderable( bucket, tile );
211 	
212 	// If no renderable on the tile, create a new renderable (or reuse an existing one)
213 	if ( !renderable )
214 	{
215 		renderable = new this.renderableConstuctor(bucket,this.tileManager.renderContext.gl);
216 		isNewRenderable = true;
217 	}
218 	
219 	if ( renderable.addGeometry( geometry, tile ) && tile.children )
220 	{
221 		// Recursively add the geometry to loaded children
222 		for ( var i = 0; i < 4; i++ )
223 		{
224 			if ( tile.children[i].state == GlobWeb.Tile.State.LOADED )
225 			{
226 				this.addGeometryToTile( bucket, geometry, tile.children[i] );
227 			}
228 		}
229 	}
230 	
231 	// 	If the renderable is new, add it to the tile
232 	if (isNewRenderable)
233 	{
234 		this.addRenderableToTile(tile,renderable);
235 	}
236 }

250 /**
251 	Dispose renderable data from tile
252  */
253 GlobWeb.TiledVectorRenderer.TileData.prototype.dispose = function()
254 {
255 	for ( var i=0; i < this.renderables.length; i++ )
256 	{
257 		this.renderables[i].dispose();
258 	}
259 	this.renderables.length = 0;
260 }

264 /**
265 	Add a renderable to the tile
266  */
267 GlobWeb.TiledVectorRenderer.prototype.addRenderableToTile = function( tile, renderable )
268 {
269 	if ( renderable.vertices.length > 0 )
270 	{
271 		if ( !tile.extension[this.id] )
272 			tile.extension[this.id] = new GlobWeb.TiledVectorRenderer.TileData();
273 		
274 		tile.extension[this.id].renderables.push( renderable );
275 	}
276 }

280 /**
281 	Generate renderable data on the tile
282  */
283 GlobWeb.TiledVectorRenderer.prototype.generate = function( tile )
284 {
285 	if ( tile.parent )
286 	{	
287 		// Only add geometry from parent tile (if any)
288 		var ls = tile.parent.extension[this.id];
289 		var ll = ls ?  ls.renderables.length : 0;
290 		for ( var i = 0; i < ll; i++ )
291 		{
292 			var parentRenderable = ls.renderables[i];
293 			var renderable = new this.renderableConstuctor(parentRenderable.bucket,this.tileManager.renderContext.gl);
294 			
295 			var parentGeometryInfos = parentRenderable.geometryInfos;
296 			for ( var j = 0; j < parentGeometryInfos.length; j++ )
297 			{
298 				renderable.addGeometry( parentGeometryInfos[j].geometry, tile );
299 			}
300 			
301 			this.addRenderableToTile(tile,renderable);
302 		}
303 	}
304 	else
305 	{
306 		// No parent tile : traverse all geometries to generate data on tile
307 		for ( var i = 0; i < this.buckets.length; i++ )
308 		{
309 			var bucket = this.buckets[i];
310 			var renderable = new this.renderableConstuctor(bucket,this.tileManager.renderContext.gl);
311 			
312 			for ( var j = 0; j < bucket.geometries.length; j++ )
313 			{
314 				renderable.addGeometry( bucket.geometries[j], tile );
315 			}
316 			
317 			this.addRenderableToTile(tile,renderable);
318 		}
319 	}
320 }

324 /**
325 	Render all redenrable on the given tiles
326  */
327 GlobWeb.TiledVectorRenderer.prototype.render = function( visibleTiles )
328 {
329 	var renderContext = this.tileManager.renderContext;
330 	var gl = renderContext.gl;
331 	
332 	var modelViewMatrix = mat4.create();
333 	
334     // Setup program
335     this.program.apply();
336 	
337 	gl.depthFunc(gl.LEQUAL);
338  	gl.uniformMatrix4fv( this.program.uniforms["projectionMatrix"], false, renderContext.projectionMatrix);
339     
340 	var currentStyle = null;
341 	
342     for (var i = 0; i < visibleTiles.length; ++i)
343     {
344 		var tile = visibleTiles[i];
345 		
346 		// If the tile is loaded and contains renderable, render them
347 		if ( tile.extension[this.id] )
348 		{
349 			mat4.multiply( renderContext.viewMatrix, tile.matrix, modelViewMatrix );
350 			gl.uniformMatrix4fv( this.program.uniforms["modelViewMatrix"], false, modelViewMatrix );
351 			gl.uniform1f( this.program.uniforms["zOffset"], tile.radius * 0.0007 );
352 			
353 			var renderables = tile.extension[this.id].renderables;
354 			for (var j=0; j < renderables.length; j++)
355 			{
356 				var renderable = renderables[j];
357 				
358 				if ( renderable.bucket.layer._visible )
359 				{
360 					if ( renderable.bucket.style != currentStyle )
361 					{
362 						currentStyle = renderable.bucket.style;
363 						gl.lineWidth( currentStyle.strokeWidth );
364 						gl.uniform4f( this.program.uniforms["color"], currentStyle.strokeColor[0], currentStyle.strokeColor[1], currentStyle.strokeColor[2], 
365 							currentStyle.strokeColor[3] * renderable.bucket.layer._opacity );
366 							
367 						// TODO : manage opacity
368 					}
369 					
370 					renderables[j].render( this.program.attributes );
371 				}
372 			}
373 		}
374 		// If the tile is not loaded, but its parent contains some renderable, render them 'clipped' to the child tile to avoid 'glitch' when zooming
375 		else if ( tile.state != GlobWeb.Tile.State.LOADED && tile.parent.extension[this.id] )
376 		{
377 			mat4.multiply( renderContext.viewMatrix, tile.parent.matrix, modelViewMatrix );
378 			gl.uniformMatrix4fv( this.program.uniforms["modelViewMatrix"], false, modelViewMatrix );
379 			gl.uniform1f( this.program.uniforms["zOffset"], tile.parent.radius * 0.0007 );
380 			
381 			var renderables = tile.parent.extension[this.id].renderables;
382 			for (var j=0; j < renderables.length; j++)
383 			{
384 				var renderable = renderables[j];
385 				if ( renderable.bucket.layer._visible )
386 				{
387 					if ( renderable.bucket.style != currentStyle )
388 					{
389 						currentStyle = renderable.bucket.style;
390 						gl.lineWidth( currentStyle.strokeWidth );
391 						gl.uniform4f( this.program.uniforms["color"], currentStyle.strokeColor[0], currentStyle.strokeColor[1], currentStyle.strokeColor[2], 
392 							currentStyle.strokeColor[3] * renderable.bucket.layer._opacity );
393 							
394 						// TODO : manage opacity
395 					}
396 					
397 					renderables[j].renderChild( this.program.attributes, tile.parentIndex );
398 				}
399 			}
400 		
401 		}
402     }
403 
404 	gl.depthFunc(gl.LESS);
405 }
