 22 /** 
 23 	@constructor
 24 	TilePool constructor
 25  */
 26 GlobWeb.TilePool = function(rc)
 27 {
 28 	// Private properties
 29 	var gl = rc.gl;
 30 	var glTextures = [];
 31 	var glBuffers = [];
 32 	var self = this;
 33 	
 34 	// Public properties
 35 	this.numCreatedTextures = 0;
 36 	this.numReusedTextures = 0;
 37 	
 38 	// Private methods
 39 
 40 	/**************************************************************************************************************/
 41 
 42 	/**
 43 		Create a new GL texture
 44 	 */
 45 	var createNewGLTexture = function(image)
 46 	{
 47 		var glTexture = gl.createTexture();
 48 		gl.bindTexture(gl.TEXTURE_2D, glTexture);
 49 		gl.texImage2D(gl.TEXTURE_2D, 0, gl.RGBA, gl.RGBA, gl.UNSIGNED_BYTE, image);
 50 		gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_MAG_FILTER, gl.LINEAR);
 51 		gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_MIN_FILTER, gl.LINEAR_MIPMAP_LINEAR);
 52 		gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_WRAP_S, gl.CLAMP_TO_EDGE);
 53 		gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_WRAP_T, gl.CLAMP_TO_EDGE);
 54 		gl.generateMipmap(gl.TEXTURE_2D);
 55 		
 56 		self.numCreatedTextures++;
 57 		
 58 		return glTexture;
 59 	}
 60 
 61 	/**************************************************************************************************************/
 62 
 63 	/**
 64 		Reuse a GL texture
 65 	 */
 66 	var reuseGLTexture = function(image)
 67 	{
 68 		var glTexture = glTextures.pop();
 69 		gl.bindTexture(gl.TEXTURE_2D, glTexture);
 70 		//gl.texSubImage2D(gl.TEXTURE_2D, 0, 0, 0, gl.RGBA, gl.UNSIGNED_BYTE, image);
 71 		gl.texImage2D(gl.TEXTURE_2D, 0, gl.RGBA, gl.RGBA, gl.UNSIGNED_BYTE, image);
 72 		gl.generateMipmap(gl.TEXTURE_2D);
 73 		
 74 		self.numReusedTextures++;
 75 		
 76 		return glTexture;
 77 	}
 78 		
 79 	// Public methods
 80 	
 81 	/**************************************************************************************************************/
 82 
 83 	/**
 84 		Create a GL texture to be used by a tile
 85 	 */
 86 	this.createGLTexture = function(image)
 87 	{
 88 		if ( glTextures.length > 0 )
 89 		{
 90 			return reuseGLTexture(image);
 91 		}
 92 		else
 93 		{
 94 			return createNewGLTexture(image);
 95 		}
 96 	};
 97 
 98 	/**************************************************************************************************************/
 99 
100 	/**
101 		Create a GL texture to be used by a tile
102 	 */
103 	this.createGLBuffer = function(vertices)
104 	{
105 		var vb;
106 		if ( glBuffers.length > 0 )
107 		{
108 			vb = glBuffers.pop();
109 		}
110 		else
111 		{
112 			vb = gl.createBuffer();
113 		}
114 		gl.bindBuffer(gl.ARRAY_BUFFER, vb);
115 		gl.bufferData(gl.ARRAY_BUFFER, vertices, gl.STATIC_DRAW);
116 		
117 		return vb;
118 	};
119 
120 	/**************************************************************************************************************/
121 
122 	/**
123 		Dispose a texture
124 	 */
125 	this.disposeGLTexture = function(texture)
126 	{
127 		glTextures.push(texture);
128 	}
129 
130 	/**************************************************************************************************************/
131 
132 	/**
133 		Dispose a texture
134 	 */
135 	this.disposeGLBuffer = function(buffer)
136 	{
137 		glBuffers.push(buffer);
138 	}
139 
140 	/**************************************************************************************************************/
141 
142 	/**
143 		Dispose all
144 	 */
145 	this.disposeAll = function()
146 	{
147 		for ( var i = 0;  i < glTextures.length; i++ ) {
148 			gl.deleteTexture( glTextures[i] );
149 		}
150 		glTextures.length = 0;
151 		
152 		for ( var i = 0;  i < glBuffers.length; i++ ) {
153 			gl.deleteBuffer( glBuffers[i] );
154 		}
155 		glBuffers.length = 0;
156 	}
157 
158 	/**************************************************************************************************************/	
159 };
