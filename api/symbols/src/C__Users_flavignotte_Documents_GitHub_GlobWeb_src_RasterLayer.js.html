 23 /** @export
 24 	@constructor
 25 	RasterLayer constructor
 26  */
 27  GlobWeb.RasterLayer = function( options )
 28 {
 29 	GlobWeb.BaseLayer.prototype.constructor.call( this, options );
 30 	
 31 	// Base properties
 32 	this.tilePixelSize = -1;
 33 	this.tiling = null;
 34 	this.numberOfLevels = -1;
 35 	this.overlay = options && options.hasOwnProperty('overlay') ? options['overlay'] : true;
 36 	this.geoBound = options['geoBound'] || null;
 37 	this.coordinates = options['coordinates'] || null;
 38 	
 39 	// Internal
 40 	this.ready = true;
 41 }
 42 
 43 /**************************************************************************************************************/
 44 
 45 GlobWeb.inherits( GlobWeb.BaseLayer,GlobWeb.RasterLayer );
 46 
 47 /**************************************************************************************************************/
 48 
 49 /** 
 50   Attach the raster layer to the globe
 51  */
 52 GlobWeb.RasterLayer.prototype._attach = function( g )
 53 {
 54 	if ( !this.overlay )
 55 	{
 56 		// Override id of background layer because of unicity of background not overlayed layer
 57 		this.id = 0;
 58 	}
 59 
 60 	GlobWeb.BaseLayer.prototype._attach.call( this, g );
 61 		
 62 	if ( this.overlay )
 63 	{
 64 		// Create the renderer if needed
 65 		if ( !g.rasterOverlayRenderer )
 66 		{
 67 			var renderer = new GlobWeb.RasterOverlayRenderer(g.tileManager);
 68 			g.tileManager.addPostRenderer(renderer);
 69 			g.rasterOverlayRenderer = renderer;
 70 		}
 71 		g.rasterOverlayRenderer.addOverlay(this);
 72 	}
 73 }
 74 
 75 /**************************************************************************************************************/
 76 
 77 /** 
 78   Detach the raster layer from the globe
 79  */
 80 GlobWeb.RasterLayer.prototype._detach = function()
 81 {
 82 	// Remove raster from overlay renderer if needed
 83 	if ( this.overlay && this.globe.rasterOverlayRenderer )
 84 	{
 85 		this.globe.rasterOverlayRenderer.removeOverlay(this);
 86 	}
 87 	
 88 	GlobWeb.BaseLayer.prototype._detach.call(this);
 89 }
