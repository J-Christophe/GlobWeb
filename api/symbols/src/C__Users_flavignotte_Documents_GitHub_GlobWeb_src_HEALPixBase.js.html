 20 /** @constructor
 21 	GlobWeb.HEALPixBase module
 22 	Module which contains all the maths stuff
 23 */
 24 GlobWeb.HEALPixBase = (function(){
 25 	
 26 	var HALF_PI = 3.14159265/2;
 27 	
 28 	var compress_bits = function(v){
 29 
 30 		//  raw  = v & 0x5555555555555 in place of raw = v & 0x5555555555555555
 31 		//		--> still not resolved, dunno why
 32 		//
 33 		
 34 		// in Java implementation mask == 0x5555555555555555
 35 		// var raw = v & 0x5555555555555; // v & 101010101010101010101010101010101010101010101010101010101010101
 36 										  // // raw>>>15 = 0101010101010101010101010101010101010101010101010
 37 		// var dec = raw>>>15;
 38 		// raw |= dec;				  // 101010101010101111111111111111111111111111111111111111111111111
 39 		// var raw1 = (raw&0xffff);
 40 		// var dec2 = raw>>>31;
 41 		// var raw2 = (dec2&0xffff);
 42 		
 43 		var longV = GlobWeb.Long.fromNumber(v);
 44 		var longMask = GlobWeb.Long.fromNumber(0x5555555555555);
 45 		var raw = longV.and(longMask);
 46 		var dec = raw.shiftRightUnsigned(15);
 47 		raw = raw.or(dec);
 48 		var raw1 = (raw.and(GlobWeb.Long.fromNumber(0xffff))).toInt();
 49 		var dec2 = raw.shiftRightUnsigned(32);
 50 		var raw2 = (dec2.and(GlobWeb.Long.fromNumber(0xffff))).toInt();
 51 		
 52 		return GlobWeb.HealPixTables.ctab[raw1&0xff] | (GlobWeb.HealPixTables.ctab[raw1>>>8]<< 4) | (GlobWeb.HealPixTables.ctab[raw2&0xff]<<16) | (GlobWeb.HealPixTables.ctab[raw2>>>8]<<20);
 53 	}
 54 
 55 	var spread_bits = function(v)
 56 	{
 57 		return (GlobWeb.HealPixTables.utab[ v      &0xff])      | ((GlobWeb.HealPixTables.utab[(v>>> 8)&0xff])<<16)
 58 			| ((GlobWeb.HealPixTables.utab[(v>>>16)&0xff])<<32) | ((GlobWeb.HealPixTables.utab[(v>>>24)&0xff])<<48);
 59 	}
 60 
 61 	/**
 62 	 *	Function describing a location on the sphere
 63 	*/
 64 	var fxyf = function(_fx,_fy,_face){
 65 		
 66 		var jr = GlobWeb.HealPixTables.jrll[_face] - _fx - _fy;
 67 		var z = 0;
 68 		var phi = 0;
 69 		var sth = 0;
 70 		var have_sth = false;
 71 
 72 		var nr;
 73 		if (jr<1){
 74 			nr = jr;
 75 			var tmp = nr*nr/3.;
 76 			z = 1 - tmp;
 77 			if (z>0.99) { sth=Math.sqrt(tmp*(2.-tmp)); have_sth=true; }
 78 		} else if (jr>3){
 79 			nr = 4-jr;
 80 			var tmp = nr*nr/3.;
 81 			z = tmp - 1;
 82 			if (z<-0.99) {
 83 				sth=Math.sqrt(tmp*(2.-tmp)); 
 84 				have_sth=true;
 85 			}
 86 		} else {
 87 			nr = 1;
 88 			z = (2-jr)*2./3.;
 89 		}
 90 
 91 		var tmp=GlobWeb.HealPixTables.jpll[_face]*nr+_fx-_fy;
 92 		if (tmp<0) tmp+=8;
 93 		if (tmp>=8) tmp-=8;
 94 		
 95 		phi = (nr<1e-15) ? 0 : (0.5*HALF_PI*tmp)/nr;
 96 		
 97 		var st = (have_sth) ? sth : Math.sqrt((1.0-z)*(1.0+z));
 98 		return [st*Math.cos(phi), st*Math.sin(phi), z]
 99 	}
100 	
101 	/**
102 	*	Static function
103 	*	Convert nside to order
104 	*	(ilog2(nside))
105 	*/
106 	var nside2order = function(arg){
107 		
108 		var res=0;
109 		while (arg > 0x0000FFFF) { res+=16; arg>>>=16; }
110 		if (arg > 0x000000FF) { res|=8; arg>>>=8; }
111 		if (arg > 0x0000000F) { res|=4; arg>>>=4; }
112 		if (arg > 0x00000003) { res|=2; arg>>>=2; }
113 		if (arg > 0x00000001) { res|=1; }
114 		return res;
115 	
116 	}

118 	/**
119 	 *	Returns pixel index of point on sphere
120 	 *
121 	 *	@param order Tile order
122 	 *	@param lon Longitude
123 	 *	@param lat Latitude
124 	 */
125 	var lonLat2pix = function(order, lon, lat)
126 	{
127 		var loc = lonLat2ang( lon, lat );
128 		return loc2pix( order, loc[0], loc[1] );
129 	}
130 
131 	/**
132 	 *	Longitude/latitude to HEALPix coordinate system angle conversion
133 	 *
134 	 *	@param	lon Longitude
135 	 *	@param 	lat Latitude
136 	 *	@return	{Float[]} Returns[ phi, theta ], where:
137      *				Phi is the azimuth(or azimuthal angle) which belongs to interval [0;2*pi] (counterclockwise).
138 	 *				Theta is the inclination(or polar angle) which belongs to interval [0;pi] (0 rad = north pole). 
139 	 */
140 	var lonLat2ang = function(lon, lat)
141     {
142     	if ( lon < 0 )
143     		lon += 360;
144 
145     	var phi = lon * Math.PI / 180.;
146     	
147     	var theta = ( -lat + 90 ) * Math.PI / 180;
148     	return [phi, theta];
149     }
150 
151 	/**
152 	 *	Returns the remainder of the division {@code v1/v2}.
153 	 *  The result is non-negative.
154 	 *
155 	 *	@param 	v1 dividend; can be positive or negative
156 	 *	@param 	v2 divisor; must be positive
157 	 *	@return Remainder of the division; positive and smaller than {@code v2}
158 	 */
159 	var fmodulo = function(v1, v2)
160 	{
161 		if (v1>=0)
162 			return (v1<v2) ? v1 : v1%v2;
163 		var tmp=v1%v2+v2;
164 		return (tmp==v2) ? 0. : tmp;
165 	}
166 
167 	var xyf2nest = function(ix, iy, face_num, order)
168     {
169 	    return ((face_num)<<(2*order)) +
170 	     		 spread_bits(ix) + (spread_bits(iy)<<1);
171     }
172 
173   	/**
174   	 *	Returns the pixel index of point on sphere
175   	 *	(TODO Maybe make conversation to Long for high orders(really high orders > 28))
176   	 *
177   	 *	@param 	order Tile order
178   	 *	@param	phi Azimuth(or azimuthal angle) in radians
179   	 *	@param  theta Inclination(or polar angle) in radians
180   	 *	@return The pixel index of pointing.
181 	 */
182 	var loc2pix = function(order, phi, theta)
183 	{
184 		var nside = Math.pow(2, order);
185 		var z = Math.cos(theta);
186 		var phi = phi;
187 
188 		var loc = {
189 			phi: phi,
190 			theta: theta,
191 			z: z
192 		}
193 
194 		if (Math.abs(z)>0.99)
195 		{
196 		  loc.sth = Math.sin(theta);
197 		  loc.have_sth=true;
198 		}
199 
200 		var inv_halfpi = 2/Math.PI;
201 		var tt = fmodulo((phi*inv_halfpi),4.0);// in [0,4)
202 
203 		var za = Math.abs(z);
204 		if (za<=2./3.) // Equatorial region
205 		{
206 			// Double
207 			var temp1 = nside*(0.5+tt);
208 			var temp2 = nside*(z*0.75);
209 			// Long
210 			var jp = (temp1-temp2); // index of  ascending edge line
211 			var jm = (temp1+temp2); // index of descending edge line
212 			var ifp = jp >>> order;  // in {0,4}
213 			var ifm = jm >>> order;
214 			var face_num = (ifp==ifm) ? (ifp|4) : ((ifp<ifm) ? ifp : (ifm+8));
215 
216 			var ix = jm & (nside-1),
217 				iy = nside - (jp & (nside-1)) - 1;
218 			return xyf2nest(Math.floor(ix),Math.floor(iy), Math.floor(face_num), order);
219 		}
220 			else // polar region, za > 2/3
221 		{
222 		var ntt = Math.min(3,Math.floor(tt)); // int
223 		var tp = tt-ntt; // double
224 		var tmp = ((za<0.99)||(!loc.have_sth)) ? // double
225 					nside*Math.sqrt(3*(1-za)) :
226 					nside*loc.sth/Math.sqrt((1.+za)/3.);
227 
228 		var jp = (tp*tmp); // increasing edge line index (long)
229 		var jm = ((1.0-tp)*tmp); // decreasing edge line index (long)
230 		if (jp>=nside) jp = nside-1; // for points too close to the boundary
231 		if (jm>=nside) jm = nside-1;
232 		return (z>=0) ?
233 		  xyf2nest(Math.floor(nside-jm -1),Math.floor(nside-jp-1),ntt,order) :
234 		  xyf2nest(Math.floor(jp),Math.floor(jm),ntt+8,order);
235 		}
236 	}
237 	
238 	return {
239 			compress_bits: compress_bits,
240 			fxyf: fxyf,
241 			nside2order: nside2order,
242 			lonLat2pix: lonLat2pix
243 		};
244 }());
