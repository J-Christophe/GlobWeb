 19 /** @constructor
 20 	GlobWeb.HEALPixBase module
 21 	Module which contains all the maths stuff
 22 */
 23 GlobWeb.HEALPixBase = (function(){
 24 	
 25 	var HALF_PI = 3.14159265/2;
 26 	
 27 	var compress_bits = function(v){
 28 
 29 		//  raw  = v & 0x5555555555555 in place of raw = v & 0x5555555555555555
 30 		//		--> still not resolved, dunno why
 31 		//
 32 		
 33 		// in Java implementation mask == 0x5555555555555555
 34 		// var raw = v & 0x5555555555555; // v & 101010101010101010101010101010101010101010101010101010101010101
 35 										  // // raw>>>15 = 0101010101010101010101010101010101010101010101010
 36 		// var dec = raw>>>15;
 37 		// raw |= dec;				  // 101010101010101111111111111111111111111111111111111111111111111
 38 		// var raw1 = (raw&0xffff);
 39 		// var dec2 = raw>>>31;
 40 		// var raw2 = (dec2&0xffff);
 41 		
 42 		var longV = GlobWeb.Long.fromNumber(v);
 43 		var longMask = GlobWeb.Long.fromNumber(0x5555555555555);
 44 		var raw = longV.and(longMask);
 45 		var dec = raw.shiftRightUnsigned(15);
 46 		raw = raw.or(dec);
 47 		var raw1 = (raw.and(GlobWeb.Long.fromNumber(0xffff))).toInt();
 48 		var dec2 = raw.shiftRightUnsigned(32);
 49 		var raw2 = (dec2.and(GlobWeb.Long.fromNumber(0xffff))).toInt();
 50 		
 51 		return GlobWeb.HealPixTables.ctab[raw1&0xff] | (GlobWeb.HealPixTables.ctab[raw1>>>8]<< 4) | (GlobWeb.HealPixTables.ctab[raw2&0xff]<<16) | (GlobWeb.HealPixTables.ctab[raw2>>>8]<<20);
 52 	}
 53 	
 54 	var fxyf = function(_fx,_fy,_face){
 55 		
 56 		var jr = GlobWeb.HealPixTables.jrll[_face] - _fx - _fy;
 57 		var z = 0;
 58 		var phi = 0;
 59 		var sth = 0;
 60 		var have_sth = false;
 61 
 62 		var nr;
 63 		if (jr<1){
 64 			nr = jr;
 65 			var tmp = nr*nr/3.;
 66 			z = 1 - tmp;
 67 			if (z>0.99) { sth=Math.sqrt(tmp*(2.-tmp)); have_sth=true; }
 68 		} else if (jr>3){
 69 			nr = 4-jr;
 70 			var tmp = nr*nr/3.;
 71 			z = tmp - 1;
 72 			if (z<-0.99) {
 73 				sth=Math.sqrt(tmp*(2.-tmp)); 
 74 				have_sth=true;
 75 			}
 76 		} else {
 77 			nr = 1;
 78 			z = (2-jr)*2./3.;
 79 		}
 80 
 81 		var tmp=GlobWeb.HealPixTables.jpll[_face]*nr+_fx-_fy;
 82 		if (tmp<0) tmp+=8;
 83 		if (tmp>=8) tmp-=8;
 84 		
 85 		phi = (nr<1e-15) ? 0 : (0.5*HALF_PI*tmp)/nr;
 86 		
 87 		var st = (have_sth) ? sth : Math.sqrt((1.0-z)*(1.0+z));
 88 		return [st*Math.cos(phi), st*Math.sin(phi), z]
 89 	}
 90 	
 91 	/**
 92 	*	Static function
 93 	*	Convert nside to order
 94 	*	(ilog2(nside))
 95 	*	
 96 	*/
 97 	var nside2order = function(arg){
 98 		
 99 		var res=0;
100 		while (arg > 0x0000FFFF) { res+=16; arg>>>=16; }
101 		if (arg > 0x000000FF) { res|=8; arg>>>=8; }
102 		if (arg > 0x0000000F) { res|=4; arg>>>=4; }
103 		if (arg > 0x00000003) { res|=2; arg>>>=2; }
104 		if (arg > 0x00000001) { res|=1; }
105 		return res;
106 	
107 	}
108 	
109 	return {
110 			compress_bits: compress_bits,
111 			fxyf: fxyf,
112 			nside2order: nside2order
113 		};
114 }());
