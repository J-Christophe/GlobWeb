 31 /** 
 32 	Generate the tiles for level zero
 33  */
 34 GlobWeb.HEALPixTiling.prototype.generateLevelZeroTiles = function( config, tilePool )
 35 {	
 36 	config.skirt = false;
 37 	config.cullSign = -1;
 38 	config.tesselation = 5;
 39 
 40 	var level0Tiles = [];
 41 	
 42 	var qpf = Math.pow(this.nside,2); // quad per face
 43 	var nFaces = 12;
 44 	var nQuads = nFaces * qpf;
 45 	
 46 	for (var i = 0; i < nQuads; i++){
 47 		var face = Math.floor(i/qpf);
 48 		var tile = new GlobWeb.HEALPixTile(this.order, i, face);
 49 		tile.config = config;
 50 		level0Tiles.push( tile );
 51 		tile.generate(tilePool);
 52 		tile.state = GlobWeb.Tile.State.NONE;
 53 	}
 54 
 55 	return level0Tiles;
 56 }

 74 /** @constructor
 75 	Tile constructor
 76 	
 77 		Quadrilateral which composes one pixel of HEALPix sphere
 78 		
 79 		nside : 2^order
 80 		order : log2(nside);
 81 		pix : pixel index number
 82 		face : face number = [0..11]
 83  */
 84 GlobWeb.HEALPixTile = function( order, pix, face )
 85 {
 86     // Call ancestor constructor
 87     GlobWeb.Tile.prototype.constructor.call(this);
 88 	
 89 	this.order = order;
 90 	this.nside = Math.pow(2, this.order);
 91 	this.pixelIndex = pix;
 92 	this.face = face;
 93 
 94 	// Compute texture transform
 95 	var width = 1728/64;
 96 	var height = 1856/64;
 97 	this.texTransform = [64/1728, 64/1856, ((this.pixelIndex % width))/width, ((Math.floor(this.pixelIndex/width))/height)];
 98 }

107 /**
108 	Create the children
109  */
110 GlobWeb.HEALPixTile.prototype.createChildren = function()
111 {
112 	// Create the children
113 	
114 	var child00 = new GlobWeb.HEALPixTile(this.order + 1, this.pixelIndex*4, this.face);
115 	var child10 = new GlobWeb.HEALPixTile(this.order + 1, this.pixelIndex*4+2, this.face);
116 	var child01 = new GlobWeb.HEALPixTile(this.order + 1, this.pixelIndex*4+1, this.face);
117 	var child11 = new GlobWeb.HEALPixTile(this.order + 1, this.pixelIndex*4+3, this.face);
118 	
119 	child00.initFromParent( this, 0, 0 );
120 	child10.initFromParent( this, 1, 0 );
121 	child01.initFromParent( this, 0, 1 );
122 	child11.initFromParent( this, 1, 1 );
123 	
124 	this.children = [ child00, child10, child01, child11 ];
125 
126 }

130 /**
131 	Compute the local matrix for the tile
132  */
133 GlobWeb.HEALPixTile.prototype.computeLocalMatrix = function(vertices){
134 	var matrix = mat4.create();
135 	
136 	var east = vec3.create();
137 	var north = vec3.create();
138 	var up = vec3.create();
139 	
140 	var mx = 0;
141 	var my = 0;
142 	var mz = 0;
143 	for(var i=0; i<vertices.length; i++){
144 		mx+=vertices[i][0];
145 		my+=vertices[i][1];
146 		mz+=vertices[i][2];
147 	}
148 	var barycenter = vec3.create([mx/vertices.length,my/vertices.length,mz/vertices.length]);
149 	
150 	vec3.set(barycenter,up);
151 	vec3.normalize(up);
152 	
153 	vec3.subtract(vertices[0],vertices[3],north);
154 	
155 	vec3.cross(up, north, east);
156 	vec3.normalize(east);
157 	vec3.cross(up, east, north);
158 	vec3.normalize(north);
159 		
160 	matrix[0] = east[0];
161 	matrix[1] = east[1];
162 	matrix[2] = east[2];
163 	matrix[3] = 0.0;
164 	
165 	matrix[4] = north[0];
166 	matrix[5] = north[1];
167 	matrix[6] = north[2];
168 	matrix[7] = 0.0;
169 	
170 	matrix[8] = up[0];
171 	matrix[9] = up[1];
172 	matrix[10] = up[2];
173 	matrix[11] = 0.0;
174 	
175 	matrix[12] = barycenter[0];
176 	matrix[13] = barycenter[1];
177 	matrix[14] = barycenter[2];
178 	matrix[15] = 1.0;
179 	
180 	return matrix;
181 }

185 /**
186 	Generate vertices for tile
187  */
188 GlobWeb.HEALPixTile.prototype.generateVertices = function()
189 {
190 	// Build the vertices
191 	var size = this.config.tesselation;
192 	var worldSpaceVertices = new Array();
193 	var step = 1./(size - 1);
194 	
195 	// xyf calculation
196 	//var xyf = new healpixBase.Xyf(this.pixelIndex, this.order);
197 	var pix=this.pixelIndex&(this.nside*this.nside-1);
198 	var ix = GlobWeb.HEALPixBase.compress_bits(pix);
199 	var iy = GlobWeb.HEALPixBase.compress_bits(pix>>>1);
200 	var face = (this.pixelIndex>>>(2*this.order));
201 	
202 	// Compute array of worldspace coordinates
203 	for(var u = 0; u < size; u++){
204 		for(var v = 0; v < size; v++){
205 			worldSpaceVertices[u*size + v] = GlobWeb.HEALPixBase.fxyf((ix+u*step)/this.nside, (iy+v*step)/this.nside, face);
206 		}
207 	}
208 	
209 	// Compute tile matrix
210 	this.matrix = this.computeLocalMatrix(worldSpaceVertices);	
211 	var invMatrix = mat4.create();
212 	mat4.inverse( this.matrix, invMatrix );
213 	this.inverseMatrix = invMatrix;
214 	
215 	// Compute tile matrix
216 	/*var center = GlobWeb.HEALPixBase.fxyf((ix+0.5)/this.nside, (iy+0.5)/this.nside, face);
217 	var geoCenter = GlobWeb.CoordinateSystem.from3DToGeo(center);
218 	this.matrix = GlobWeb.CoordinateSystem.getLHVTransform( geoCenter );
219 	var invMatrix = mat4.create();
220 	mat4.inverse( this.matrix, invMatrix );
221 	this.inverseMatrix = invMatrix;*/
222 	
223 	// Build the vertices
224 	var vertices = new Float32Array( 3*size*size );
225 	
226 	// Vertex coordinates in local space
227 	var vertexOffset = 0;
228 	for(var i=0;i<worldSpaceVertices.length;i++){
229 		vertices[vertexOffset] = invMatrix[0]*worldSpaceVertices[i][0] + invMatrix[4]*worldSpaceVertices[i][1] + invMatrix[8]*worldSpaceVertices[i][2] + invMatrix[12];
230 		vertices[vertexOffset+1] = invMatrix[1]*worldSpaceVertices[i][0] + invMatrix[5]*worldSpaceVertices[i][1] + invMatrix[9]*worldSpaceVertices[i][2] + invMatrix[13];
231 		vertices[vertexOffset+2] = invMatrix[2]*worldSpaceVertices[i][0] + invMatrix[6]*worldSpaceVertices[i][1] + invMatrix[10]*worldSpaceVertices[i][2] + invMatrix[14];
232 		vertexOffset += 3;
233 	}
234 	
235 	return vertices;
236 }
