1 /**************************************************************************************************************/
  2 
  3 /** @export
  4 	@constructor
  5 	KeyboardNavigationHandler constructor
  6 	@param options Configuration properties for the KeyboardNavigationHandler :
  7 			<ul>
  8 				<li>panFactor  : Factor for panning within the scene</li>
  9 				<li>zoomFactor : Factor for zooming into the scene</li>
 10 				<li>installOnDocument : True to install the event listener on the document and not on the canvas</li>
 11 			</ul>
 12  */
 13 GlobWeb.KeyboardNavigationHandler = function(options){
 14 	
 15 	/**************************************************************************************************************/
 16 	
 17 	/**
 18  	 * Private variables
 19 	 */
 20 	var _navigation = null;
 21 	var self = this;
 22 	
 23 	/**
 24 	 * Public variables
 25 	 */
 26 	this.panFactor = 10.;
 27 	this.zoomFactor = 1.;
 28 	
 29 	// Setup options
 30 	if ( options )
 31 	{
 32 		if ( options['panFactor'] && typeof options['panFactor'] == 'number' )
 33 			this.panFactor = options['panFactor'];
 34 		if ( options['zoomFactor'] && typeof options['zoomFactor'] == 'number' )
 35 			this.zoomFactor = options['zoomFactor'];
 36 	}
 37 	
 38 	/**************************************************************************************************************/
 39 	
 40 	/**
 41  	 * Private methods
 42 	 */
 43 
 44 	/**
 45 	 * Set focus
 46 	 */
 47 	var _setFocus = function(event)
 48 	{
 49 		this.focus();
 50 		return false;
 51 	};
 52 	  
 53 	/**
 54 	 *	Event handler for key down
 55 	 */
 56 	var _handleKeyDown = function(event)
 57 	{
 58 		switch( event.keyCode ){
 59 			case 33 :
 60 				// Page Up
 61 				_navigation.zoom(-self.zoomFactor);
 62 				break;
 63 			case 34 :
 64 				// Page Down
 65 				_navigation.zoom(self.zoomFactor);
 66 				break;
 67 			case 37 :
 68 				// Left arrow
 69 				if ( event.ctrlKey )
 70 				{
 71 					_navigation.rotate( -self.panFactor, 0 );
 72 				}
 73 				else
 74 				{
 75 					_navigation.pan( self.panFactor, 0 );
 76 				}
 77 				break;
 78 			case 38 :
 79 				// Up arrow
 80 				_navigation.pan( 0, self.panFactor );
 81 				break;
 82 			case 39 :
 83 				// Right arrow
 84 				if ( event.ctrlKey )
 85 				{
 86 					_navigation.rotate( self.panFactor, 0 );
 87 				}
 88 				else
 89 				{
 90 					_navigation.pan( -self.panFactor, 0 );
 91 				}
 92 				break;
 93 			case 40 :
 94 				// Down arrow
 95 				_navigation.pan( 0, -self.panFactor );
 96 				break;
 97 		}
 98 		_navigation.globe.renderContext.requestFrame();
 99 	};
100 
101 	/**************************************************************************************************************/
102 	
103 	 /**
104 	  * Public methods
105 	  */
106 
107 	/** 
108 		Setup the default event handlers for the navigation
109 	 */
110 	this.install = function(navigation)
111 	{
112 		// Setup the keyboard event handlers
113 		_navigation = navigation;
114 		
115 		if ( options && options.installOnDocument )
116 		{
117 			document.addEventListener("keydown", _handleKeyDown);
118 		}
119 		else
120 		{
121 			var canvas = _navigation.globe.renderContext.canvas;
122 			canvas.addEventListener("keydown", _handleKeyDown);
123 			// Setup focus handling to receive keyboard event on canvas
124 			canvas.tabIndex = "0";
125 			canvas.addEventListener("mousedown", _setFocus);
126 		}
127 	};
128 
129 	/** 
130 		Remove the default event handlers for the navigation
131 	 */
132 	this.uninstall = function()
133 	{	
134 		if ( options && options.installOnDocument )
135 		{
136 			document.removeEventListener("keydown", _handleKeyDown);
137 		}
138 		else
139 		{
140 			var canvas = _navigation.globe.renderContext.canvas;
141 			canvas.removeEventListener("keydown", _handleKeyDown);
142 			canvas.removeEventListener("mousedown", _setFocus);
143 		}
144 	};
145 	
146 };
