 22 /** @export
 23 	@constructor
 24   PathAnimation is an animation defined with a path.
 25  */
 26 GlobWeb.PathAnimation = function(coords,speed,valueSetter)
 27 {
 28     // Call ancestor constructor
 29     GlobWeb.Animation.prototype.constructor.call(this);
 30 
 31     this.speed = speed * GlobWeb.CoordinateSystem.heightScale / 1000;
 32 	this.nodes = [];
 33 	for ( var i = 0; i < coords.length; i++ )
 34 	{
 35 		var node = { position: GlobWeb.CoordinateSystem.fromGeoTo3D(coords[i]),
 36 					velocity: null,
 37 					distance: 0.0 };
 38 		this.nodes.push( node );
 39 		if ( i > 0 )
 40 		{
 41 			var dx = this.nodes[i].position[0] - this.nodes[i-1].position[0];
 42 			var dy = this.nodes[i].position[1] - this.nodes[i-1].position[1];
 43 			var dz = this.nodes[i].position[2] - this.nodes[i-1].position[2];
 44 			this.nodes[i-1].distance = Math.sqrt( dx*dx + dy*dy + dz*dz );
 45 		}
 46 	}
 47 	
 48 	for ( var i = 1; i <  coords.length - 1; i++ )
 49 	{
 50 		var vec1 = vec3.subtract( this.nodes[i+1].position, this.nodes[i].position, vec3.create() );
 51 		var vec2 = vec3.subtract( this.nodes[i-1].position, this.nodes[i].position, vec3.create() );
 52 		vec3.normalize(vec1);
 53 		vec3.normalize(vec2);
 54 		this.nodes[i].velocity = vec3.subtract( vec1, vec2, vec3.create() );
 55 		vec3.normalize(this.nodes[i].velocity);
 56 	}
 57 	
 58 	// Start velocity
 59 	var temp = vec3.subtract( this.nodes[1].position, this.nodes[0].position, vec3.create() );
 60 	vec3.scale( temp, ( 3 / this.nodes[0].distance ) );
 61 	this.nodes[0].velocity = vec3.subtract( temp, this.nodes[1].velocity, vec3.create() );
 62 	vec3.scale( this.nodes[0].velocity, 0.5 );
 63 	
 64 	// End velocity
 65 	var i = coords.length - 1;
 66 	var temp = vec3.subtract( this.nodes[i].position, this.nodes[i-1].position, vec3.create() );
 67 	vec3.scale( temp, ( 3 / this.nodes[i-1].distance ) );
 68 	this.nodes[i].velocity = vec3.subtract( temp, this.nodes[i-1].velocity, vec3.create() );
 69 	vec3.scale( this.nodes[i].velocity, 0.5 );
 70 
 71 	this.index = 0;
 72 	this.currentDistance = 0;
 73 	this.previousTime = -1;
 74 	this.currentDirection = [];
 75 	this.centerOffset = -0.2;
 76 	this.altitudeOffset = 1000;
 77 	
 78 	var that = this;
 79 	if ( valueSetter )
 80 	{
 81 		this.valueSetter = valueSetter;
 82 	}
 83 	else
 84 	{
 85 		this.valueSetter = function(value,direction)
 86 		{				
 87 			var up = vec3.normalize( value, vec3.create() );
 88 			
 89 			var geoEye = GlobWeb.CoordinateSystem.from3DToGeo( value );
 90 			geoEye[2] = that.globe.getElevation( geoEye[0], geoEye[1] ) + that.altitudeOffset;
 91 			var eye =  GlobWeb.CoordinateSystem.fromGeoTo3D( geoEye );
 92 			
 93 			//var eye = vec3.add( vec3.scale(up, (that.altitudeOffset+elevation) * GlobWeb.CoordinateSystem.heightScale, vec3.create()), value );
 94 			var dirn = vec3.normalize( direction, vec3.create() );
 95 			var center = vec3.add( eye, dirn, vec3.create() );
 96 			vec3.add( center, vec3.scale(up, that.centerOffset, vec3.create()) );
 97 			mat4.lookAt( eye, center, up, that.globe.renderContext.viewMatrix );
 98 		};
 99 	}
100 }

108 /**
109 	Set the speed
110  */
111 GlobWeb.PathAnimation.prototype.setSpeed = function(val)
112 {
113     this.speed = parseFloat(val) * GlobWeb.CoordinateSystem.heightScale / 1000;
114 }
115 
116 /**************************************************************************************************************/
117 
118 /**
119 	Set the altitude offset
120  */
121 GlobWeb.PathAnimation.prototype.setAltitudeOffset = function(val)
122 {
123 	this.altitudeOffset = parseFloat(val);
124 }

126 /**************************************************************************************************************/

128 /**
129 	Set the direction angle
130  */
131 GlobWeb.PathAnimation.prototype.setDirectionAngle = function(vertical)
132 {
133 	this.centerOffset = Math.tan( parseFloat(vertical) * Math.PI / 180.0 );
134 }

138 /** @export
139 	Start the animation
140  */
141 GlobWeb.PathAnimation.prototype.start = function()
142 {
143 	var previousStartTime = -1;
144 	if ( this.pauseTime != -1 )
145 	{
146 		previousStartTime = this.startTime;
147 	}
148 
149     GlobWeb.Animation.prototype.start.call(this);
150 	
151 	if ( previousStartTime != -1 )
152 	{
153 		this.previousTime += this.startTime - previousStartTime;
154 	}
155 	else
156 	{
157 		this.previousTime = -1;
158 	}
159 }

163 /*
164 	Animation update method
165 */
166 GlobWeb.PathAnimation.prototype.update = function(now)
167 {
168 	if ( this.previousTime == -1 )
169 	{
170 		this.index = 0;
171 		this.currentDistance = 0;
172 	}
173 	else
174 	{
175 		this.currentDistance += (now - this.previousTime) * this.speed;
176 	}
177 	this.previousTime = now;
178 
179 	while ( this.currentDistance >= this.nodes[this.index].distance && this.index < this.nodes.length - 1 )
180 	{
181 		this.currentDistance -= this.nodes[this.index].distance;
182 		this.index = this.index + 1;
183 	}
184 	
185 	if ( this.index < this.nodes.length - 1 )
186 	{
187 		var t = this.currentDistance / this.nodes[this.index].distance;
188 		var startPos = this.nodes[this.index].position;
189 		var endPos = this.nodes[this.index+1].position;
190 		var startVel = vec3.scale( this.nodes[this.index].velocity, this.nodes[this.index].distance, vec3.create() );
191 		var endVel = vec3.scale( this.nodes[this.index+1].velocity, this.nodes[this.index].distance, vec3.create() );
192 		var position = Numeric.cubicInterpolation( t, startPos, startVel, endPos, endVel );
193 		var direction = Numeric.cubicInterpolationDerivative( t, startPos, startVel, endPos, endVel );
194 		this.valueSetter( position, direction );
195 	}
196 	else if ( this.index == this.nodes.length - 1 )
197 	{
198 		this.valueSetter( this.nodes[this.index].position, this.nodes[this.index].velocity );
199 	}
200 	else
201 	{
202 		this.stop();
203 	}
204 }
