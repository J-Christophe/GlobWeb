 23 /** @constructor
 24  *  @extends GlobWeb.TiledVectorRenderable
 25  *	LineStringRenderable manages lineString data to be rendered on a tile.
 26  */
 27 GlobWeb.LineStringRenderable = function( bucket, gl )
 28 {
 29 	GlobWeb.TiledVectorRenderable.prototype.constructor.call(this,bucket,gl);
 30 	this.glMode = gl.LINES;
 31 }

 40 /**
 41  * Build children indices.
 42  * Children indices are used to render a tile children when it is not completely loaded.
 43  */
 44 GlobWeb.LineStringRenderable.prototype.buildChildrenIndices = function()
 45 {
 46 	this.childrenIndices = [ [], [], [], [] ];
 47 	this.childrenIndexBuffers = [ null, null, null, null ];
 48 		
 49 	for ( var n = 0;  n < this.indices.length/2; n++ )
 50 	{	
 51 		var vertexOffset1 = 3 * this.indices[2*n];
 52 		var vertexOffset2 = 3 * this.indices[2*n+1];
 53 		
 54 		var x1 = this.vertices[vertexOffset1];
 55 		var x2 = this.vertices[vertexOffset2];
 56 		
 57 		var i = 0;
 58 		if ( x1 > 0 ||  ( x1 == 0 && x2 > 0 ) )
 59 			i = 1;			
 60 		
 61 		var y1 = this.vertices[vertexOffset1+1];
 62 		var y2 = this.vertices[vertexOffset2+1];
 63 		
 64 		var j = 1;
 65 		if ( y1 > 0 ||  ( y1 == 0 && y2 > 0 ) )
 66 			j = 0;
 67 		
 68 		this.childrenIndices[ 2*j + i ].push( this.indices[2*n] )
 69 		this.childrenIndices[ 2*j + i ].push( this.indices[2*n+1] )
 70 	}
 71 }

 75 /**
 76  * Build vertices and indices from the coordinates.
 77  * Clamp a line string on a tile
 78  */
 79 GlobWeb.LineStringRenderable.prototype.buildVerticesAndIndices = function( tile, coords )
 80 {
 81 	if ( coords.length == 0 )
 82 		return;
 83 
 84 	// Fix date line for coordinates first
 85 	var coordinates = this._fixDateLine( tile, coords );
 86 		
 87 	var size = tile.config.tesselation;
 88 	var vs = tile.config.vertexSize;
 89 	
 90 	// Convert lon/lat coordinates to tile coordinates (between [0,size-1] inside the tile)
 91 	var tileCoords = tile.lonlat2tile(coords);
 92 
 93 	for ( var i = 0; i < coordinates.length - 1; i++ )
 94 	{
 95 		var u1 = tileCoords[i][0];
 96 		var v1 = tileCoords[i][1];
 97 		
 98 		var u2 = tileCoords[i+1][0];
 99 		var v2 = tileCoords[i+1][1];
100 		
101 		var intersections = [];
102 	
103 		// Intersect the segment with the tile grid
104 		
105 		// First intersect with columns
106 		// uStart, uEnd represent a range of the tile columns that the segement can intersect
107 		var uStart = Math.max( -1, Math.min( u1, u2 ) );
108 		var uEnd = Math.min( size-1, Math.max( u1, u2 ) );
109 		for ( var n = Math.floor(uStart)+1; n < Math.floor(uEnd)+1; n++)
110 		{
111 			var u = n;
112 			var res = Numeric.lineIntersection( u1, v1, u2, v2, u, 0.0, u, size-1 );
113 			if ( res[0] > 0.0 && res[0] < 1.0 && res[1] >= 0.0 && res[1] <= 1.0 )
114 			{
115 				var v = res[1] * (size-1);
116 				var vFloor = Math.floor( v );
117 				var vFrac = v - vFloor;
118 				var vertexOffset = vs*( vFloor*size + n );
119 				var x = (1.0 - vFrac) * tile.vertices[ vertexOffset ] + vFrac * tile.vertices[ vertexOffset + vs*size ];
120 				var y = (1.0 - vFrac) * tile.vertices[ vertexOffset + 1 ] + vFrac * tile.vertices[ vertexOffset + vs*size + 1 ];
121 				var z = (1.0 - vFrac) * tile.vertices[ vertexOffset + 2 ] + vFrac * tile.vertices[ vertexOffset + vs*size + 2 ];
122 				intersections.push( [ res[0], x, y, z ] );
123 			}
124 		}
125 	
126 		// Then intersect with rows
127 		// vStart, vEnd represent a range of the tile rows that the segement can intersect
128 		var vStart = Math.max( -1, Math.min( v1, v2 ) );
129 		var vEnd = Math.min( size-1, Math.max( v1, v2 ) );
130 		for ( var n = Math.floor(vStart)+1; n < Math.floor(vEnd)+1; n++)
131 		{
132 			var v = n;
133 			var res = Numeric.lineIntersection( u1, v1, u2, v2, 0.0, v, size-1, v );
134 			if ( res[0] > 0.0 && res[0] < 1.0 && res[1] >= 0.0 && res[1] <= 1.0 )
135 			{
136 				var u = res[1] * (size-1);
137 				var uFloor = Math.floor( u );
138 				var uFrac = u - uFloor;
139 				var vertexOffset = vs*( n*size + uFloor );
140 				var x = (1.0 - uFrac) * tile.vertices[ vertexOffset ] + uFrac * tile.vertices[ vertexOffset + vs ];
141 				var y = (1.0 - uFrac) * tile.vertices[ vertexOffset + 1 ] + uFrac * tile.vertices[ vertexOffset + vs+1 ];
142 				var z = (1.0 - uFrac) * tile.vertices[ vertexOffset + 2 ] + uFrac * tile.vertices[ vertexOffset + vs+2 ];
143 				intersections.push( [ res[0], x, y, z ] );
144 			}
145 		}
146 /*			for ( var n = 0; n < size; n++)
147 		{
148 			var u = n;
149 			var v = n;
150 			var res = lineIntersection( u1, v1, u2, v2, u, 0.0, 0.0, v );
151 			if ( res[0] > 0.0 && res[0] < 1.0 && res[1] > 0.0 && res[1] < 1.0 )
152 			{
153 				intersections.push( res[0] );
154 			}
155 			if ( n != size-1 )
156 			{
157 				var res = lineIntersection( u1, v1, u2, v2, u, size, size, v );
158 				if ( res[0] > 0.0 && res[0] < 1.0 && res[1] > 0.0 && res[1] < 1.0 )
159 				{
160 					intersections.push( res[0] );
161 				}
162 			}
163 		}*/
164 		
165 		// Sort intersections found on the segment
166 		intersections.sort( function(a,b) { return a[0] > b[0]; } );
167 		
168 		// Build the vertices from the intersections found
169 		var startIndex = this.vertices.length / 3;
170 		
171 		if ( u1 >= 0.0 && u1 <= size-1 &&  v1 >= 0.0 && v1 <= size-1 )
172 		{
173 			var vec = tile.computePosition(u1,v1);
174 			this.vertices.push( vec[0] );
175 			this.vertices.push( vec[1] );
176 			this.vertices.push( vec[2] );
177 		}
178 					
179 		for ( var n = 0; n < intersections.length; n++ )
180 		{
181 			this.vertices.push( intersections[n][1] );
182 			this.vertices.push( intersections[n][2] );
183 			this.vertices.push( intersections[n][3] );
184 		}
185 		
186 		if ( u2 >= 0.0 && u2 <= size-1 &&  v2 >= 0.0 && v2 <= size-1 )
187 		{
188 			var vec = tile.computePosition(u2,v2);
189 			this.vertices.push( vec[0] );
190 			this.vertices.push( vec[1] );
191 			this.vertices.push( vec[2] );
192 		}
193 		
194 		var endIndex = this.vertices.length / 3;
195 		
196 		for ( var n = startIndex; n < endIndex - 1; n++ )
197 		{
198 			this.indices.push( n );
199 			this.indices.push( n+1 );
200 		}
201 	}
202 }

206 // Register the renderer
207 GlobWeb.VectorRendererManager.registerRenderer({
208 					creator: function(globe) { 
209 						var lineStringRenderer = new GlobWeb.TiledVectorRenderer(globe.tileManager);
210 						lineStringRenderer.id = "lineString";
211 						lineStringRenderer.styleEquals = function(s1,s2) { return s1.isEqualForLine(s2); };
212 						lineStringRenderer.renderableConstuctor = GlobWeb.LineStringRenderable;
213 						return lineStringRenderer;
214 					},
215 					canApply: function(type,style) {
216 						// LineStringRenderer supports line string (multi or not) and polygon (or multi) when not filled
217 
218 						return (style.rendererHint == "Tiled") && (type == "LineString" || type == "MultiLineString"
219 							|| (!style.fill && (type == "Polygon" || type == "MultiPolygon"))); 
220 					} 
221 				});
