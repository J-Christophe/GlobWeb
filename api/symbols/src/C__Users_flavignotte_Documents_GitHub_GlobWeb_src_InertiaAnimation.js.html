 22 /**	@constructor
 23  *	Animation simulating inertia for camera navigation
 24  *
 25  *	@param nav Navigation
 26  */
 27 GlobWeb.InertiaAnimation = function(nav)
 28 {
 29     GlobWeb.Animation.prototype.constructor.call(this);
 30 
 31 	this.factor = 0.95;
 32 	this.type = null;
 33 	this.dx = 0;
 34 	this.dy = 0;
 35 	this.navigation = nav;
 36 	this.navigation.globe.addAnimation(this);
 37 }
 38 
 39 /**************************************************************************************************************/
 40 
 41 GlobWeb.inherits(GlobWeb.Animation,GlobWeb.InertiaAnimation);
 42 
 43 /**************************************************************************************************************/
 44 
 45 GlobWeb.InertiaAnimation.prototype.update = function(now)
 46 {
 47 	if ( this.factor > 0 )
 48 	{		
 49 		switch(this.type)
 50 		{
 51 			case "pan":
 52 				this.navigation.pan(this.dx,this.dy);
 53 				break;
 54 			case "rotate":
 55 				this.navigation.rotate(this.dx,this.dy);
 56 				break;
 57 			case "zoom":
 58 				this.navigation.zoom(this.dx);
 59 				break;
 60 			default:
 61 		}
 62 		this.dx *= this.factor;
 63 		if (this.dy) this.dy *= this.factor;
 64 		this.navigation.globe.renderContext.requestFrame();
 65 	}
 66 	else
 67 	{
 68 		this.stop();
 69         return;
 70 	}
 71 }
 72 
 73 /**************************************************************************************************************/
 74 
 75 /**
 76  *	@param type Type of inertia
 77  *				<ul>
 78  *					<li>pan</li>
 79  *					<li>rotate</li>
 80  *					<li>zoom</li>
 81  *				</ul>
 82  *	@param speed Starting speed
 83  *	@param {Int[]} inertiaVector Vector of mouvement in window coordinates(for pan and rotate inertias)
 84  */
 85 GlobWeb.InertiaAnimation.prototype.launch = function(type, dx, dy)
 86 {
 87 	// Set first value
 88  	this.type = type;
 89 	this.dx = dx;
 90 	this.dy = dy;
 91 
 92 	this.start();
 93 }
