  3 /** @export
  4 	@constructor
  5 	BaseNavigation constructor
  6  */
  7 GlobWeb.BaseNavigation = function(globe, options)
  8 {
  9 	this.globe = globe;
 10 
 11 	// Copy options
 12 	for (var x in options)
 13 	{
 14 		this[x] = options[x];
 15 	}
 16 	
 17 	// Create default handlers if none are created in options
 18 	if ( !this.handlers ) 
 19 	{
 20 		this.handlers = [new GlobWeb.MouseNavigationHandler({ zoomOnDblClick : true }), new GlobWeb.KeyboardNavigationHandler()];
 21 	}
 22 
 23 	// Install handlers
 24 	for (var i=0; i<this.handlers.length; i++)
 25 	{
 26 		this.handlers[i].install(this);
 27 	}
 28 }

 32 /** @export
 33 	Get the field of view used by the navigation
 34 	
 35 	@return {Float[]} Fovx and fovy in degrees
 36 */
 37 
 38 GlobWeb.BaseNavigation.prototype.getFov = function()
 39 {
 40 	var aspect = this.globe.renderContext.canvas.width / this.globe.renderContext.canvas.height;
 41 	return [ aspect * this.globe.renderContext.fov, this.globe.renderContext.fov ];
 42 }
