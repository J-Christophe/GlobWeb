 23 /** @export
 24 	@constructor
 25 	WCSElevationLayer constructor
 26  */
 27 GlobWeb.WCSElevationLayer = function( options )
 28 {
 29 	GlobWeb.RasterLayer.prototype.constructor.call( this, options );
 30 	
 31 	this.baseUrl = options['baseUrl'];
 32 	this.tilePixelSize = options['tilePixelSize'] || 33;
 33 	this.tiling = new GlobWeb.GeoTiling( 4, 2 );
 34 	this.numberOfLevels = options['numberOfLevels'] || 21;
 35 	this.type = "ImageryRaster";
 36 	this.version = options['version'] || '2.0.0';
 37 	this.format = options['format'] || 'image/x-aaigrid';
 38 	
 39 	// Build the base GetMap URL
 40 	var url = this.baseUrl;
 41 	if ( url.indexOf('?',0) == -1 )
 42 	{
 43 		url += '?service=wcs';
 44 	}
 45 	else
 46 	{
 47 		url += '&service=wcs';
 48 	}
 49 	url += "&version=" + this.version;
 50 	url += "&request=GetCoverage";
 51 
 52 	switch (this.version.substring(0,3)) 
 53 	{
 54 		case '2.0':
 55 			this.crs = options['outputCRS'] || options['crs'] || 'http://www.opengis.net/def/crs/EPSG/0/4326';
 56 			url += '&outputCRS=' + this.crs;
 57 			url += "&size=x(" + this.tilePixelSize + ")";
 58 			url += "&size=y(" + this.tilePixelSize + ")";
 59 			url += "&coverageid=" + options['coverage'];
 60 			break;
 61 		case '1.0':
 62 			url += "&width=" + this.tilePixelSize;
 63 			url += "&height=" + this.tilePixelSize;
 64 			url += '&crs=' + (options['crs'] || 'EPSG:4326');
 65 			url += "&coverage=" + options['coverage'];
 66 			break;
 67 	}
 68 	url += '&format=' + this.format;
 69 	
 70 	this.getCoverageBaseUrl = url;
 71 }
 72 
 73 GlobWeb.inherits(GlobWeb.RasterLayer,GlobWeb.WCSElevationLayer);

 78 /**
 79 	Parse a elevation response
 80  */
 81 GlobWeb.WCSElevationLayer.prototype.parseElevations = function(text)
 82 {
 83 	if (text == null) {
 84 		return this._returnZeroElevations();
 85 	}
 86 	switch(this.format) {
 87 	case "image/x-aaigrid":
 88 		return this._parseAAIGrid(text);
 89 	default:
 90 		console.log("Format '" + this.format + "' could not be parsed.");
 91 		return this._returnZeroElevations();
 92 	}
 93 }

 98 /**
 99 	Fallback elevations when no data was returned
100  */
101 GlobWeb.WCSElevationLayer.prototype._returnZeroElevations = function()
102 {
103 	var elevations = [];
104 	for( var i = 0; i < this.tilePixelSize * this.tilePixelSize; ++i ) {
105 		elevations.push(0);
106 	}
107 	return elevations;
108 }

114 /**
115 	Parse a elevation response from AAIGrid
116  */
117 GlobWeb.WCSElevationLayer.prototype._parseAAIGrid = function(text)
118 {
119 	var elevations = [];
120 	var lines = text.trim().split('\n');
121 
122 	var dataLinesStart = 0;
123 	for ( var i = 0; i < lines.length; ++i ) {
124 		if (lines[i].substring(0, 1) === " ") {
125 			dataLinesStart = i;
126 			break;
127 		}
128 	}
129 
130 	for ( var i = dataLinesStart; i < lines.length; i++ )
131 	{
132 		var elts = lines[i].trim().split(/\s+/);
133 		for ( var n=0; n < elts.length; n++ )
134 		{
135 			elevations.push( parseInt(elts[n]) );
136 		}
137 	}
138 
139 	return elevations;
140 }

145 /**
146 	Get an url for the given tile
147  */
148 GlobWeb.WCSElevationLayer.prototype.getUrl = function(tile)
149 {
150 	var geoBound = tile.geoBound;
151 	var url = this.getCoverageBaseUrl;
152 
153 	if (this.version.substring(0,3) === '2.0') 
154 	{
155 		url += '&subset=x,' + this.crs + '(' + geoBound.west + ',' + geoBound.east + ')';
156 		url += '&subset=y,' + this.crs + '(' + geoBound.south + ',' + geoBound.north + ')';
157 	}
158 	else if (this.version.substring(0,3) === '1.0') 
159 	{
160 		url += "&bbox=";	
161 		url += geoBound.west;
162 		url += ",";
163 		url += geoBound.south;
164 		url += ",";
165 		url += geoBound.east;
166 		url += ",";
167 		url += geoBound.north;
168 	}
169 	
170 	return url;
171 }
