 22 /** @constructor
 23 	Tile constructor
 24  */
 25 GlobWeb.Tile = function()
 26 {
 27 	// Parent/child relationship
 28 	this.parent = null;
 29 	this.parentIndex = -1;
 30 	this.children = null;
 31 	
 32 	// Graphics data to render the tile
 33 	this.vertices = null;
 34 	this.texture = null;
 35 	this.vertexBuffer = null;
 36 	this.texTransform = [1., 1., 0., 0.];
 37 	
 38 	// Tile spatial data
 39 	this.matrix = null;
 40 	this.inverseMatrix = null;
 41 	this.bbox = new GlobWeb.BoundingBox();
 42 	
 43 	// For culling
 44 	this.radius = 0.0;	
 45 	this.distance = 0.0;
 46 	this.closestPointToEye = [ 0.0, 0.0, 0.0 ];
 47 	
 48 	// Specific object to store extension from renderers
 49 	this.extension = {};
 50 	
 51 	// For debug
 52 	//this.color = [ Math.random(), Math.random(), Math.random() ];
 53 	
 54 	this.state = GlobWeb.Tile.State.NONE;
 55 	
 56 	// Tile configuration given by tile manager : contains if the tile uses skirt, the tesselation, etc...
 57 	this.config = null;
 58 }

 77 /**
 78  * Compute position on the tile using normalized coordinate between [0,size-1]
 79  */
 80 GlobWeb.Tile.prototype.computePosition = function(u,v)
 81 {
 82 	var vFloor = Math.floor( v );
 83 	var vFrac = v - vFloor;
 84 	var uFloor = Math.floor( u );
 85 	var uFrac = u - uFloor;
 86 	var size = this.config.tesselation;
 87 	var vertexSize = this.config.vertexSize;
 88 	var vertexOffset = vertexSize*( vFloor*size + uFloor );
 89 	var vec = [ 0.0, 0.0, 0.0 ];
 90 	for ( var i=0; i < 3; i++)
 91 	{
 92 		vec[i] = (1.0 - vFrac) * (1.0 - uFrac) * this.vertices[ vertexOffset + i ]
 93 		+ vFrac * (1.0 - uFrac) * this.vertices[ vertexOffset + vertexSize*size + i ]
 94 		+ vFrac * uFrac * this.vertices[ vertexOffset + vertexSize*size + vertexSize + i ]
 95 		+ (1.0 - vFrac) * uFrac * this.vertices[ vertexOffset + vertexSize + i ];
 96 	}
 97 	
 98 	return vec;
 99 }

104 /**
105  *	Initialize the tile from its parent
106  */
107 GlobWeb.Tile.prototype.initFromParent = function(parent,i,j)
108 {
109 	this.parent = parent;
110 	this.parentIndex = j*2 + i;
111 	this.matrix  = parent.matrix;
112 	this.inverseMatrix  = parent.inverseMatrix;
113 	this.texture = parent.texture;
114 	this.config = parent.config;
115 	
116 	this.vertexBuffer = parent.vertexBuffer;
117 	
118 	// Recompute the bounding box
119 	// Very fast and coarse version but it does not work with HEALPix tiling
120 	//var w = 0.5 * (parent.bbox.max[0] - parent.bbox.min[0]);
121 	//var h = -0.5 * (parent.bbox.max[1] - parent.bbox.min[1]);
122 	//var min = [  parent.bbox.min[0] + i * w, parent.bbox.max[1] + (j+1) * h, parent.bbox.min[2] ];
123 	//var max = [  parent.bbox.min[0] + (i+1) * w, parent.bbox.max[1] + j * h, parent.bbox.max[2] ];
124 	
125 	var size = this.config.tesselation;
126 	var halfTesselation = (size-1)/2;
127 	for (var n = 0; n <= halfTesselation; n++)
128 	{
129 		var offset = this.config.vertexSize * ( (n+j*halfTesselation)*size + i*halfTesselation );
130 		for (var k = 0; k <= halfTesselation; k++)
131 		{
132 			this.bbox.extend( parent.vertices[offset], parent.vertices[offset+1], parent.vertices[offset+2] );
133 			offset += this.config.vertexSize;
134 		}
135 	}
136 	
137 	// Compute the bounding box
138 	this.radius = this.bbox.getRadius();
139 }

143 /**
144  *	Test if the tile needs to be refined
145  */
146 GlobWeb.Tile.prototype.needsToBeRefined = function(renderContext)
147 {
148 	if ( this.distance < this.radius )
149 		return true;
150 
151 	// Approximate the radius of one texel : the radius of the tile divided by the image size
152 	var radius = this.radius / this.config.imageSize; 
153 	
154 	// Transform the closest point from the eye in world coordinates
155 	var mat = this.matrix;
156 	var c = this.closestPointToEye;
157 	var px = mat[0]*c[0] + mat[4]*c[1] + mat[8]*c[2] + mat[12];
158 	var py = mat[1]*c[0] + mat[5]*c[1] + mat[9]*c[2] + mat[13];
159 	var pz = mat[2]*c[0] + mat[6]*c[1] + mat[10]*c[2] + mat[14];
160 	
161 	// Compute the pixel size of the radius texel
162 	var pixelSizeVector = renderContext.pixelSizeVector;
163 	var pixelSize = radius / ( px * pixelSizeVector[0] + py * pixelSizeVector[1]
164 						+ pz * pixelSizeVector[2] + pixelSizeVector[3] );
165 	
166 	// Check if pixel radius of a texel is superior to the treshold
167 	return pixelSize > renderContext.tileErrorTreshold;
168 }

172 /**
173  *	Test if the tile is culled given the current view parameters
174  */
175 GlobWeb.Tile.prototype.isCulled = function(renderContext)
176 {	
177 	// Compute the eye in tile local space
178 	var mat = this.inverseMatrix;
179 	var c = renderContext.eyePosition;
180 	var ex = mat[0]*c[0] + mat[4]*c[1] + mat[8]*c[2] + mat[12];
181 	var ey = mat[1]*c[0] + mat[5]*c[1] + mat[9]*c[2] + mat[13];
182 	var ez = mat[2]*c[0] + mat[6]*c[1] + mat[10]*c[2] + mat[14];
183 			
184 	// If the eye is in the radius of the tile, consider the tile is not culled
185 	var distance = Math.sqrt( ex * ex + ey * ey + ez * ez );
186 	if ( distance < this.radius )
187 	{
188 		this.distance = 0.0;
189 		renderContext.near = renderContext.minNear;
190 		renderContext.far = Math.max( renderContext.far, 2.0 * this.radius );
191 		return false;
192 	}
193 	else
194 	{
195 		var pt = this.closestPointToEye;
196 		
197 		// Compute closest point to eye with the bbox of the tile
198 		pt[0] = Math.min( Math.max( ex, this.bbox.min[0] ), this.bbox.max[0] );
199 		pt[1] = Math.min( Math.max( ey, this.bbox.min[1] ), this.bbox.max[1] );
200 		pt[2] = Math.min( Math.max( ez, this.bbox.min[2] ), this.bbox.max[2] );
201 				
202 		/*// Compute eye direction at the closest point
203 		var edx = ex - pt[0];
204 		var edy = ey - pt[1];
205 		var edz = ez - pt[2];*/
206 		
207 		// Transform eye direction in tile local space
208 		var ed = renderContext.eyeDirection;
209 		var edx = -(mat[0]*ed[0] + mat[4]*ed[1] + mat[8]*ed[2]);
210 		var edy = -(mat[1]*ed[0] + mat[5]*ed[1] + mat[9]*ed[2]);
211 		var edz = -(mat[2]*ed[0] + mat[6]*ed[1] + mat[10]*ed[2]);
212 		
213 		// Compute vertical at the closest point. The earth center is [0, 0, -radius] in tile local space.
214 		var vx = pt[0];
215 		var vy = pt[1];
216 		var vz = pt[2] + GlobWeb.CoordinateSystem.radius;
217 		
218 		// Compute dot product between eye direction and the vertical at the point
219 		//var el = Math.sqrt( edx * edx + edy * edy  + edz * edz );
220 		var vl = Math.sqrt( vx * vx + vy * vy + vz * vz );
221 		//var eDv = (edx * vx + edy * vy  + edz * vz) / ( vl * el );
222 		var eDv = (edx * vx + edy * vy  + edz * vz) / vl;
223 		
224 		// Update distance, distance is used when requesting a tile, closer tile are loaded first
225 		//this.distance = el;
226 		this.distance = distance;
227 		
228 		eDv *= this.config.cullSign;
229 		
230 		if ( eDv < -0.05 )
231 		{
232 			return true;
233 		}
234 		else
235 		{
236 			// Compute local frustum
237 			var localFrustum = renderContext.localFrustum;
238 			localFrustum.inverseTransform( renderContext.worldFrustum, this.matrix );
239 			
240 			// Check if the tile is inside the frustum
241 			if ( localFrustum.containsBoundingBox(this.bbox) )			
242 			{
243 				// Update near/far to take into account the tile
244 				//renderContext.near = Math.max( renderContext.minNear, Math.min( renderContext.near, distance - 2.0 * this.radius ) );
245 				//renderContext.far = Math.max( renderContext.far, distance + 2.0 * this.radius );
246 				var closestDistance = -(edx * pt[0] + edy * pt[1] + edz * pt[2] - (edx * ex + edy * ey + edz * ez));
247 				renderContext.near = Math.max( renderContext.minNear, Math.min( renderContext.near, closestDistance ) );
248 				renderContext.far = Math.max( renderContext.far, closestDistance + 2.0 * this.radius );
249 				return false;
250 			}
251 			else
252 			{
253 				return true;
254 			}
255 		}
256 	}
257 }

261 /**
262  *	Dispose the tile
263  */
264 GlobWeb.Tile.prototype.dispose = function(renderContext,tilePool)
265 {		
266 	if ( this.state == GlobWeb.Tile.State.LOADED  )
267 	{
268 		tilePool.disposeGLBuffer(this.vertexBuffer);
269 		tilePool.disposeGLTexture(this.texture);
270 		
271 		for ( var x in this.extension )
272 		{
273 			if ( this.extension[x].dispose )
274 				this.extension[x].dispose(renderContext,tilePool);
275 		}
276 
277 		this.vertexBuffer = null;
278 		this.texture = null;
279 		this.parent = null;
280 		
281 		this.state = GlobWeb.Tile.State.NONE;
282 	}
283 }

287 /**
288  *	Delete the children
289  */
290 GlobWeb.Tile.prototype.deleteChildren = function(renderContext,tilePool)
291 {
292 	if ( this.children )
293 	{
294 		// Dispose children resources, and then delete its children
295 		for (var i = 0; i < 4; i++)
296 		{
297 			this.children[i].dispose(renderContext,tilePool);
298 			this.children[i].deleteChildren(renderContext,tilePool);
299 		}
300 		
301 		// Cleanup the tile
302 		this.children = null;
303 	}
304 }

308 /**
309  *	Build skirt vertices
310  */
311 GlobWeb.Tile.prototype.buildSkirtVertices = function(center,srcOffset,srcStep,dstOffset)
312 {
313 	var vertices = this.vertices;
314 	var skirtHeight = this.radius * 0.05;
315 	
316 	var size = this.config.tesselation;
317 	for ( var i = 0; i < size; i++)
318 	{
319 /*		//Not optimized version of skirt computation
320 		var srcPos = [ vertices[srcOffset], vertices[srcOffset+1], vertices[srcOffset+2] ];
321 		var dir = vec3.subtract( srcPos, center, vec3.create() );
322 		vec3.normalize(dir);
323 		vec3.scale( dir, skirtHeight );
324 		vec3.subtract( srcPos, dir );*/
325 		
326 		// Optimized version of skirt computation
327 		var x = vertices[srcOffset] - center[0];
328 		var y = vertices[srcOffset+1] - center[1];
329 		var z = vertices[srcOffset+2] - center[2];
330 		var scale = skirtHeight / Math.sqrt( x*x + y*y + z*z );
331 		x *= scale;
332 		y *= scale;
333 		z *= scale;
334 		
335 		vertices[ dstOffset ] = vertices[srcOffset] - x;
336 		vertices[ dstOffset+1 ] = vertices[srcOffset+1] - y;
337 		vertices[ dstOffset+2 ] = vertices[srcOffset+2] - z;
338 		
339 		for (var n = 3; n < this.config.vertexSize; n++)
340 		{
341 			vertices[ dstOffset+n ] = vertices[srcOffset+n];
342 		}
343 		
344 		dstOffset += this.config.vertexSize;
345 		srcOffset += srcStep;
346 	}	
347 }

351 /**
352  *	Generate normals for a tile
353  */
354 GlobWeb.Tile.prototype.generateNormals = function()
355 {	
356 	var size = this.config.tesselation;
357 	var vertexSize = this.config.vertexSize;
358 	var lineSize = vertexSize*size;
359 	
360 	var vo = 0;
361 	for ( var j=0; j < size; j++ )
362 	{
363 		var vp1 = j == size-1 ? 0 : lineSize;
364 		var vm1 = j == 0 ? 0 : -lineSize;
365 		for ( var i=0; i < size; i++ )
366 		{
367 			var up1 = i == size-1 ? 0 : vertexSize;
368 			var um1 = i == 0 ? 0 : -vertexSize;
369 			var u = [
370 				this.vertices[vo+up1] - this.vertices[vo+um1],
371 				this.vertices[vo+up1+1] - this.vertices[vo+um1+1],
372 				this.vertices[vo+up1+2] - this.vertices[vo+um1+2],
373 			];
374 			var v = [
375 				this.vertices[vo+vp1] - this.vertices[vo+vm1],
376 				this.vertices[vo+vp1+1] - this.vertices[vo+vm1+1],
377 				this.vertices[vo+vp1+2] - this.vertices[vo+vm1+2],
378 			];
379 			
380 			var normal = vec3.cross( u, v, [] );
381 			vec3.normalize(normal);
382 			this.vertices[vo+3] = normal[0];
383 			this.vertices[vo+4] = normal[1];
384 			this.vertices[vo+5] = normal[2];
385 			
386 			vo += vertexSize;
387 		}
388 	}
389 }

393 /**
394  *	Generate the tile
395  */
396 GlobWeb.Tile.prototype.generate = function(tilePool,image,elevations)
397 {
398 	// Generate the vertices
399 	this.vertices = this.generateVertices(elevations);
400 		
401 	// Compute the bounding box
402 	var size = this.config.tesselation;
403 	var vertexSize = this.config.vertexSize;
404 	this.bbox.compute(this.vertices,vertexSize*size*size,vertexSize);
405 	this.radius = this.bbox.getRadius();
406 	
407 	// Compute normals if needed
408 	if (this.config.normals)
409 	{
410 		this.generateNormals();
411 	}
412 		
413 	// Compute skirt from vertices
414 	if (this.config.skirt)
415 	{
416 		// Compute local earth center, used to generate skirts
417 		var localEarthCenter = [ 0.0, 0.0, 0.0 ];
418 		mat4.multiplyVec3( this.inverseMatrix, localEarthCenter );
419 		
420 		// Skirts
421 		var dstOffset = vertexSize * (size * size); // TOP
422 		this.buildSkirtVertices( localEarthCenter, 0, vertexSize, dstOffset );
423 		dstOffset += vertexSize * size; // BOTTOM
424 		this.buildSkirtVertices( localEarthCenter, vertexSize * (size * (size-1)), vertexSize, dstOffset );
425 		dstOffset += vertexSize * size; // LEFT
426 		this.buildSkirtVertices( localEarthCenter, 0, vertexSize * size, dstOffset );
427 		dstOffset += vertexSize * size; // RIGHT
428 		this.buildSkirtVertices( localEarthCenter, vertexSize * (size-1), vertexSize * size, dstOffset );
429 		
430 		// These skirts are only used by children tile
431 		dstOffset += vertexSize * size; // CENTER
432 		this.buildSkirtVertices( localEarthCenter, vertexSize * ( size * (size-1)/2 ), vertexSize, dstOffset );
433 		dstOffset += vertexSize * size; // MIDDLE
434 		this.buildSkirtVertices( localEarthCenter, vertexSize * ( (size-1)/2 ), vertexSize * size, dstOffset );
435 	}	
436 	
437 	// Avoid double creation of vertex buffer for level0Tiles generation
438 	if (this.vertexBuffer != null && this.parent == null)
439 	{
440 		tilePool.disposeGLBuffer(this.vertexBuffer);
441 	}
442 	this.vertexBuffer = tilePool.createGLBuffer(this.vertices);
443 
444 	// Create texture
445 	if (image)
446 	{
447 		this.texture = tilePool.createGLTexture(image);
448 	}
449 	
450 	this.state = GlobWeb.Tile.State.LOADED;
451 }
