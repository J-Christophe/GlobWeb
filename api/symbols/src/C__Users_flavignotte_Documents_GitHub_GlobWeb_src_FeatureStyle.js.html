 22 /** @constructor
 23 	FeatureStyle construtor
 24  */
 25 GlobWeb.FeatureStyle = function(style)
 26 {
 27 	// Color used for lines or polygon outline
 28 	this.strokeColor = [1.0, 0.0, 0.0, 1.0];
 29 	// Color used to full polygon
 30 	this.fillColor = [1.0, 0.0, 0.0, 1.0];
 31 	this.fillTextureUrl = null;
 32 	this.strokeWidth = 1;
 33 	this.iconUrl = "hotspot.png";
 34 	this.icon = null;
 35 	this.label = null;
 36 	this.textColor = [1.0, 1.0, 1.0, 1.0];
 37 	this.fill = false;
 38 	this.pointMaxSize = 40;
 39 	this.opacity = 1.;
 40 	this['rendererHint'] = "Tiled";
 41 	
 42 	if ( style )
 43 	{
 44 		for ( var s in style )
 45 		{
 46 			this[s] = style[s];
 47 		}
 48 	}
 49 }

206 var parseHex = /^(\w{2})(\w{2})(\w{2})$/;
207 var parseRgb = /^rgb\((\d{1,3}),\s*(\d{1,3}),\s*(\d{1,3})\)$/;
208 var parseRgba = /^rgba\((\d{1,3}),\s*(\d{1,3}),\s*(\d{1,3},\s*(\d{1,3}))\)$/;
209 
210 var parseColorString = function(color_string)
211 {
212         var r = 0, g = 0, b = 0, a = 255;
213 		var match;
214 		
215         color_string = color_string.trim();
216         color_string = color_string.toLowerCase();
217         // strip any leading #
218         if (color_string.charAt(0) == '#') { // remove # if any
219            color_string = color_string.substr(1,6);
220         }
221 		
222 		// Convert a litteral color to rgb string
223 		if ( simple_colors.hasOwnProperty(color_string) )
224 		{
225 			color_string = simple_colors[color_string];
226 		}
227 		
228 		match = parseHex.exec(color_string);
229 		if ( match )
230 		{
231 			r = parseInt(match[1],16);
232 			g = parseInt(match[2],16);
233 			b = parseInt(match[3],16);
234 		}
235 		
236 		match = parseRgb.exec(color_string);
237 		if ( match )
238 		{
239 			r = parseInt(match[1]);
240 			g = parseInt(match[2]);
241 			b = parseInt(match[3]);
242 		}
243 		
244 		match = parseRgba.exec(color_string);
245 		if ( match )
246 		{
247 			r = parseInt(match[1]);
248 			g = parseInt(match[2]);
249 			b = parseInt(match[3]);
250 			a = parseInt(match[4]);
251 		}
252 	
253         // validate/cleanup values
254         r = (r < 0) ? 0 : ((r > 255) ? 255 : r);
255         g = (g < 0) ? 0 : ((g > 255) ? 255 : g);
256         b = (b < 0) ? 0 : ((b > 255) ? 255 : b);
257         a = (a < 0) ? 0 : ((a > 255) ? 255 : a);
258 
259         return [r / 255.0, g / 255.0, b / 255.0, a / 255.0];
260 }
261 
262 return parseColorString;
263 
264 })();

268 /** 
269  * Convert an internal color to a string based color representation
270  */
271 GlobWeb.FeatureStyle.fromColorToString = function(color)
272 {		
273 	var rr = parseInt( color[0] * 255.0 ).toString(16);
274 	var gg = parseInt( color[1] * 255.0 ).toString(16);
275 	var bb = parseInt( color[2] * 255.0 ).toString(16);
276 
277 	return '#' + rr + gg + bb;
278 }

283  * Check if a style is equals to render poly
284  */
285 GlobWeb.FeatureStyle.prototype.isEqualForPoly = function(style)
286 {
287 	return this.fill == style.fill;
288 }

292 /** 
293  * Check if a style is equals to render poly
294  */
295 GlobWeb.FeatureStyle.prototype.isEqualForLine = function(style)
296 {
297 	return this.strokeColor[0] == style.strokeColor[0]
298 		&& this.strokeColor[1] == style.strokeColor[1]
299 		&& this.strokeColor[2] == style.strokeColor[2]
300 		&& this.strokeColor[3] == style.strokeColor[3]
301 		&& this.strokeWidth == style.strokeWidth;
302 }

306 /** 
307  * Check if a style is equals to render point
308  */
309 GlobWeb.FeatureStyle.prototype.isEqualForPoint = function(style)
310 {
311 	return this.iconUrl == style.iconUrl
312 		&& this.icon == style.icon
313 		&& this.label == style.label;
314 }
