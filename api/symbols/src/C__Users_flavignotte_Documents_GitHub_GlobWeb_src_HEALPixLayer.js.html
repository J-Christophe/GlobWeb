 21 /** @export
 22 	@constructor
 23 	HEALPixLayer constructor
 24 */
 25 
 26 GlobWeb.HEALPixLayer = function(options)
 27 {
 28 	GlobWeb.RasterLayer.prototype.constructor.call( this, options );
 29 	
 30 	this.tilePixelSize = options.tilePixelSize || 512;
 31 	this.tiling = new GlobWeb.HEALPixTiling( options.baseLevel || 3 );
 32 	this.numberOfLevels = options.numberOfLevels || 10;
 33 	this.type = "ImageryRaster";
 34 	this.baseUrl = options['baseUrl'];
 35 	
 36 	// allsky
 37 	this.levelZeroImage = new Image();
 38 	var self = this;
 39 	this.levelZeroImage.crossOrigin = '';
 40 	this.levelZeroImage.onload = function () 
 41 	{
 42 		self.ready = true;
 43 		
 44 		// Call callback if set
 45 		if (options.onready && options.onready instanceof Function)
 46 		{
 47 			options.onready(self);
 48 		}
 49 		
 50 		// Request a frame
 51 		if ( self.globe )
 52 		{
 53 			self.globe.renderContext.requestFrame();
 54 		}
 55 	}
 56 	this.levelZeroImage.onerror = function(event) {
 57 		console.log("Cannot load " + self.levelZeroImage.src );
 58 	}
 59 	
 60 	this.ready = false;
 61 }
 62 
 63 /**************************************************************************************************************/
 64 
 65 GlobWeb.inherits(GlobWeb.RasterLayer,GlobWeb.HEALPixLayer);
 66 
 67 /**************************************************************************************************************/
 68 
 69 /** 
 70   Attach the raster layer to the globe
 71  */
 72 GlobWeb.HEALPixLayer.prototype._attach = function( g )
 73 {
 74 	GlobWeb.RasterLayer.prototype._attach.call( this, g );
 75 
 76 	// Load level zero image now
 77 	this.levelZeroImage.src = this.baseUrl + "/Norder3/Allsky.jpg";
 78 }

 82 /**
 83  *	Get url from a given tile
 84  */
 85 GlobWeb.HEALPixLayer.prototype.getUrl = function(tile)
 86 {
 87 	var url = this.baseUrl;
 88 	
 89 	url += "/Norder";
 90 	url += tile.order;
 91 	
 92 	url += "/Dir";
 93 	var indexDirectory = Math.floor(tile.pixelIndex/10000) * 10000;
 94 	url += indexDirectory;
 95 	
 96 	url += "/Npix";
 97 	url += tile.pixelIndex;
 98 	url += ".jpg";
 99 	
100 	return url;
101 }
