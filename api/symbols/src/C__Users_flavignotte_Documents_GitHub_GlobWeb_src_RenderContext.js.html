 22 /** 
 23 	@constructor
 24 	Function constructor for RencerContext
 25 */
 26 GlobWeb.RenderContext = function(options)
 27 {
 28 	/**
 29 	 * Private properties
 30 	 */
 31 	 
 32 	/**
 33 	 * Private method
 34 	 */
 35 
 36 	
 37 	/**
 38 	 * Constructor
 39 	 */
 40 	this.shadersPath = options['shadersPath'] || "../shaders/";
 41 	this.tileErrorTreshold = options['tileErrorTreshold'] || 4;
 42 	this.lighting = options['lighting'] || false;
 43 	this.continuousRendering = options['continuousRendering'] || false;
 44 	this.stats = null;
 45 
 46 	// Init GL
 47 	var canvas = null;
 48 	
 49 	// Check canvas options
 50 	if (!options['canvas'])
 51 		throw "GlobWeb : no canvas in options";
 52 	
 53 
 54 	if (typeof options['canvas'] == "string") 
 55 	{
 56 		canvas = document.getElementById(options['canvas']);
 57 	}
 58 	else
 59 	{
 60 		canvas = options['canvas'];
 61 	}
 62 	
 63 	// Check canvas is valid
 64 	if (!canvas instanceof HTMLCanvasElement)
 65 		throw "GlobWeb : invalid canvas";
 66 
 67 	// Create the webl context
 68 	var names = ["webgl", "experimental-webgl", "webkit-3d", "moz-webgl"];
 69 	var gl = null;
 70 	for (var ii = 0; ii < names.length && gl == null; ++ii) 
 71 	{
 72 		try 
 73 		{
 74 		  gl = canvas.getContext(names[ii], options['contextAttribs']);
 75 		} 
 76 		catch(e) {}
 77 	}
 78 	
 79 	if ( gl == null )
 80 		throw "GlobWeb : WebGL context cannot be initialized";
 81 
 82 	
 83 	if ( options['backgroundColor'] )
 84 	{
 85 		var color = options['backgroundColor'];
 86 		gl.clearColor(color[0],color[1],color[2],color[3]);
 87 	}
 88 	else
 89 	{
 90 		gl.clearColor(0.0, 0.0, 0.0, 1.0);
 91 	}
 92 	
 93 	gl.pixelStorei( gl['UNPACK_COLORSPACE_CONVERSION_WEBGL'], gl.NONE );
 94 	gl.enable(gl.DEPTH_TEST);
 95 	gl.enable(gl.CULL_FACE);
 96 		
 97 	// Store local variable into static object
 98 	this.viewMatrix = mat4.create();
 99 	this.modelViewMatrix =  mat4.create();
100 	this.projectionMatrix = mat4.create();
101 	this.gl = gl;
102 	this.canvas = canvas;
103 	this.frustum = new GlobWeb.Frustum();
104 	this.worldFrustum = new GlobWeb.Frustum();
105 	this.localFrustum = new GlobWeb.Frustum();
106 	this.eyePosition = vec3.create();
107 	this.eyeDirection = vec3.create();
108 	this.minNear = 0.00001;
109 	this.near = GlobWeb.RenderContext.minNear;
110 	this.far = 6.0;
111 	this.numActiveAttribArray = 0;
112 	this.frameRequested = false;
113 	this.fov = 45;
114 	
115 	
116 	// Initialize the window requestAnimationFrame
117 	if ( !window.requestAnimationFrame ) 
118 	{
119 		window.requestAnimationFrame = ( function() {
120 			return window.webkitRequestAnimationFrame ||
121 				 window.mozRequestAnimationFrame ||
122 				window.oRequestAnimationFrame ||
123 				window.msRequestAnimationFrame ||
124 				function( callback, element ) { window.setTimeout( callback, 1000 / 60 );};
125 			} )();
126 	}
127 }

131 /** 
132 	Request a frame
133 */
134 GlobWeb.RenderContext.prototype.requestFrame = function()
135 {	
136 	if (!this.frameRequested)
137 	{
138 		window.requestAnimationFrame(this.frame);
139 		this.frameRequested = true;
140 	}
141 }

144 /** 
145 	Update properies that depends on the view matrix
146 */
147 GlobWeb.RenderContext.prototype.updateViewDependentProperties = function()
148 {
149 	var inverseViewMatrix = mat4.create();
150 	mat4.inverse( this.viewMatrix, inverseViewMatrix );
151 	
152 	vec3.set( [ 0.0, 0.0, 0.0 ], this.eyePosition );
153 	mat4.multiplyVec3( inverseViewMatrix, this.eyePosition );
154 	
155 	vec3.set( [ 0.0, 0.0, -1.0 ], this.eyeDirection );
156 	mat4.rotateVec3( inverseViewMatrix, this.eyeDirection );
157 	
158 	this.pixelSizeVector = this.computePixelSizeVector();
159 	
160 	// Init projection matrix
161 	mat4.perspective(this.fov, this.canvas.width / this.canvas.height, this.minNear, this.far, this.projectionMatrix);
162 	
163 	// Compute the frustum from the projection matrix
164 	this.frustum.compute(this.projectionMatrix);
165 	
166 	// Compute the world frustum
167 	this.worldFrustum.inverseTransform( this.frustum, this.viewMatrix );
168 }

172 /**
173 	Get mouse coordinates relative to the canvas element
174 */
175 GlobWeb.RenderContext.prototype.getXYRelativeToCanvas = function(event)
176 {
177 	// cf. http://stackoverflow.com/questions/55677/how-do-i-get-the-coordinates-of-a-mouse-click-on-a-canvas-element
178 	var pos = [];
179 	if (event.pageX || event.pageY)
180 	{
181 		pos[0] = event.pageX;
182 		pos[1] = event.pageY;
183 	}
184 	else
185 	{ 
186 		pos[0] = event.clientX + document.body.scrollLeft + document.documentElement.scrollLeft; 
187 		pos[1] = event.clientY + document.body.scrollTop + document.documentElement.scrollTop; 
188 	}
189 	
190 	var element = this.canvas; 
191 	while (element)
192 	{
193 		pos[0] -= element.offsetLeft;
194 		pos[1] -= element.offsetTop;
195 		element = element.offsetParent;
196 	}
197         
198 	return pos;
199 }

204 /** 
205 	Compute the pixel size vector
206 */
207 GlobWeb.RenderContext.prototype.computePixelSizeVector = function( mv )
208 {
209 	// pre adjust P00,P20,P23,P33 by multiplying them by the viewport window matrix.
210 	// here we do it in short hand with the knowledge of how the window matrix is formed
211 	// note P23,P33 are multiplied by an implicit 1 which would come from the window matrix.
212 	// Robert Osfield, June 2002.
213 	
214 	var width = this.canvas.width;
215 	var height = this.canvas.height;
216 	var P = this.projectionMatrix;
217 	var V = mv || this.viewMatrix;
218 	
219 	// scaling for horizontal pixels
220 	var P00 = P[0]*width*0.5;
221 	var P20_00 = P[8]*width*0.5 + P[11]*width*0.5;
222 	var scale_00 = [ V[0]*P00 + V[2]*P20_00,
223 			V[4]*P00 + V[6]*P20_00,
224 			V[8]*P00 + V[10]*P20_00 ];
225 
226 	// scaling for vertical pixels
227 	var P10 = P[5]*height*0.5;
228 	var P20_10 = P[9]*height*0.5 + P[11]*height*0.5;
229 	var scale_10 = [ V[1]*P10 + V[2]*P20_10,
230 			V[5]*P10 + V[6]*P20_10,
231 			V[9]*P10 + V[10]*P20_10 ];
232 
233 	var P23 = P[11];
234 	var P33 = P[15];
235 	var pixelSizeVector = [V[2]*P23,
236 				V[6]*P23,
237 				V[10]*P23,
238 				V[14]*P23 + V[15]*P33];
239 
240 	var scaleRatio  = 0.7071067811 / Math.sqrt( vec3.dot(scale_00,scale_00)+ vec3.dot(scale_10,scale_10) );
241 	pixelSizeVector[0] *= scaleRatio;
242 	pixelSizeVector[1] *= scaleRatio;
243 	pixelSizeVector[2] *= scaleRatio;
244 	pixelSizeVector[3] *= scaleRatio;
245 
246 	return pixelSizeVector;
247 }

250 /** 
251 	Get 3D from a pixel
252 */
253 GlobWeb.RenderContext.prototype.get3DFromPixel = function(x,y)
254 {
255 	// reverse y because (0,0) is top left but opengl's normalized
256 	// device coordinate (-1,-1) is bottom left
257 	var nx = ((x / this.canvas.width) * 2.0) - 1.0;
258 	var ny = -(((y / this.canvas.height) * 2.0) - 1.0);
259 	
260 	var tmpMat = mat4.create();
261 	mat4.multiply(this.projectionMatrix, this.viewMatrix, tmpMat);
262 	mat4.inverse(tmpMat);
263 	// Transform pos to world using inverse viewProjection matrix
264 	var worldPick = [];
265 	mat4.multiplyVec4(tmpMat, [ nx, ny, 0, 1], worldPick);
266 	worldPick[0] /= worldPick[3];
267 	worldPick[1] /= worldPick[3];
268 	worldPick[2] /= worldPick[3];
269 	
270 	// Shoot a ray from the camera to the picked point
271 	// Get camera position
272 	mat4.inverse(this.viewMatrix, tmpMat);
273 	var worldCam = [tmpMat[12], tmpMat[13], tmpMat[14]];
274 	var rayDirection = vec3.create();
275 	vec3.subtract(worldPick, worldCam, rayDirection);
276 	vec3.normalize(rayDirection);
277 	
278 	// Intersect earth sphere
279 	var t = Numeric.raySphereIntersection(worldCam, rayDirection, [0.0, 0.0, 0.0], GlobWeb.CoordinateSystem.radius);
280 	if (t >= 0)
281 	{
282 		var pos3d = Numeric.pointOnRay(worldCam, rayDirection, t);
283 		return pos3d;
284 	}
285 	
286 	return null;
287 }

291 /** 
292 	Get pixel from 3D
293 */
294 GlobWeb.RenderContext.prototype.getPixelFrom3D = function(x,y,z)
295 {
296 	var viewProjectionMatrix = mat4.create();
297 	mat4.multiply(this.projectionMatrix, this.viewMatrix, viewProjectionMatrix);
298 	
299 	// transform world to clipping coordinates
300 	point3D = [];
301 	mat4.multiply(viewProjectionMatrix, [x,y,z,0], point3D);
302 	
303 	// transform clipping to window coordinates
304 	var winX = Math.round((( 1 + point3D[0] ) / 2.0) * this.canvas.width );
305 	
306 	// reverse y because (0,0) is top left but opengl's normalized
307 	// device coordinate (-1,-1) is bottom left
308 	var winY = Math.round((( 1 - point3D[1] ) / 2.0) * this.canvas.height );
309 
310 	return [winX, winY];
311 }

315 /** 
316 	Create a non power of two texture from an image
317 */
318 GlobWeb.RenderContext.prototype.createNonPowerOfTwoTextureFromImage = function(image)
319 {	
320 	var gl = this.gl;
321 	var tex = gl.createTexture();
322 	gl.bindTexture(gl.TEXTURE_2D, tex);
323 	gl.texImage2D(gl.TEXTURE_2D, 0, gl.RGBA, gl.RGBA, gl.UNSIGNED_BYTE, image);
324 	gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_MAG_FILTER, gl.LINEAR);
325 	gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_MIN_FILTER, gl.LINEAR);
326 	gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_WRAP_S, gl.CLAMP_TO_EDGE);
327 	gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_WRAP_T, gl.CLAMP_TO_EDGE);
328 	return tex;
329 }
