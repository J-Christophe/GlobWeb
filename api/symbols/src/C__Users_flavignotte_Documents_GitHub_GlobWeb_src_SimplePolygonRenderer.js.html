 22 /** @constructor
 23  *	SimplePolygonRenderer constructor
 24  *
 25  *	Renderer of textured or not quadrilaterals
 26  */
 27 
 28 GlobWeb.SimplePolygonRenderer = function(tileManager)
 29 {
 30 	this.renderContext = tileManager.renderContext;
 31 	var gl = this.renderContext.gl;
 32 	
 33 	this.renderables = [];
 34 	
 35 	// Texture cache management : avoid flickering when modifiying style
 36 	// TODO : maybe put this in GlobWeb, can be used by other renderers
 37 	this.textureCache = {};
 38 	this.texturesToPurge = [];
 39 	
 40 	// Parameters used to implement ONE shader for color xor texture rendering
 41 	this.whiteTexture = gl.createTexture();
 42 	gl.bindTexture(gl.TEXTURE_2D, this.whiteTexture);
 43 	var whitePixel = new Uint8Array([255, 255, 255, 255]);
 44 	gl.texImage2D(gl.TEXTURE_2D, 0, gl.RGBA, 1, 1, 0, gl.RGBA, gl.UNSIGNED_BYTE, whitePixel);
 45 	
 46 	var vertexShader = "\
 47 	attribute vec3 vertex;\n\
 48 	attribute vec2 tcoord;\n\
 49 	uniform mat4 viewProjectionMatrix;\n\
 50 	\n\
 51 	varying vec2 vTextureCoord;\n\
 52 	\n\
 53 	void main(void) \n\
 54 	{\n\
 55 		vTextureCoord = tcoord;\n\
 56 		gl_Position = viewProjectionMatrix * vec4(vertex, 1.0);\n\
 57 	}\n\
 58 	";
 59 
 60 	var fragmentShader = "\
 61 	precision highp float; \n\
 62 	uniform vec4 u_color;\n\
 63 	varying vec2 vTextureCoord;\n\
 64 	uniform sampler2D texture; \n\
 65 	void main(void)\n\
 66 	{\n\
 67 		gl_FragColor = texture2D(texture, vTextureCoord) * u_color;\n\
 68 	}\n\
 69 	";
 70 	
 71 	this.program = new GlobWeb.Program(this.renderContext);
 72 	this.program.createFromSource(vertexShader, fragmentShader);
 73 
 74 	// Shared buffer
 75 	// Create texCoord buffer
 76 	this.tcoordBuffer = gl.createBuffer();
 77 	gl.bindBuffer(gl.ARRAY_BUFFER, this.tcoordBuffer);
 78 	
 79 	var textureCoords = [
 80 		0.0, 0.0,
 81 		1.0, 0.0,
 82 		1.0, 1.0,
 83 		0.0, 1.0,
 84 		0.0, 0.0
 85 	];
 86 	
 87 	gl.bufferData(gl.ARRAY_BUFFER, new Float32Array(textureCoords), gl.STATIC_DRAW);
 88 	this.tcoordBuffer.itemSize = 2;
 89 	this.tcoordBuffer.numItems = 5;
 90 }

 94 /**
 95  *	Add polygon to renderer
 96  */
 97 GlobWeb.SimplePolygonRenderer.prototype.addGeometry = function(geometry, layer, style){
 98 	
 99 	var gl = this.renderContext.gl;
100 	
101 	// Create renderable
102 	var renderable = {
103 		geometry : geometry,
104 		style : style,
105 		layer: layer,
106 		vertexBuffer : gl.createBuffer(),
107 		indexBuffer : gl.createBuffer(),
108 		texture : null,
109 		textureUrl: ""
110 	}
111 	
112 	// Create texture
113 	var self = this;
114 	
115 	if ( style.fillTextureUrl )
116 	{
117 		var cacheTexture = this.textureCache[style.fillTextureUrl];
118 		if ( cacheTexture )
119 		{
120 			renderable.texture = cacheTexture.texture;
121 			cacheTexture.count++;
122 		}
123 		else
124 		{
125 			var image = new Image();
126 			image.crossOrigin = '';
127 			image.onload = function () 
128 			{
129 				renderable.texture = self.renderContext.createNonPowerOfTwoTextureFromImage(image);
130 				self.textureCache[style.fillTextureUrl] = { texture: renderable.texture, count: 1 };
131 			}
132 			
133 			image.onerror = function(event)
134 			{
135 				console.log("Cannot load " + image.src );
136 			}
137 			
138 			image.src = style.fillTextureUrl;
139 		}
140 		renderable.textureUrl = style.fillTextureUrl;
141 	}
142 	
143 	// Create vertex buffer
144 	gl.bindBuffer(gl.ARRAY_BUFFER, renderable.vertexBuffer);
145 	var vertices = [];
146 	var points = [];
147 	
148 	// For polygons only
149 	for ( var i=0; i<geometry['coordinates'][0].length; i++)
150 	{
151 		var pos3d = [];
152 		GlobWeb.CoordinateSystem.fromGeoTo3D(geometry['coordinates'][0][i], pos3d);
153 		vertices = vertices.concat(pos3d);
154 		points.push(pos3d);
155 	}
156 
157 	gl.bufferData(gl.ARRAY_BUFFER, new Float32Array(vertices), gl.STATIC_DRAW);
158 	renderable.vertexBuffer.itemSize = 3;
159 	renderable.vertexBuffer.numItems = vertices.length/3;
160 
161 	// Create index buffer(make shared ?)
162 	var indices = [];
163 	indices = GlobWeb.Triangulator.process( points );
164 	
165 	if ( indices == null )
166 	{
167 		console.err("Triangulation error ! Check if your GeoJSON geometry is valid");
168 		return false;
169 	}
170 	gl.bindBuffer(gl.ELEMENT_ARRAY_BUFFER, renderable.indexBuffer);
171 	gl.bufferData(gl.ELEMENT_ARRAY_BUFFER, new Uint16Array(indices), gl.STATIC_DRAW);
172 	renderable.indexBuffer.itemSize = 1;
173 	renderable.indexBuffer.numItems = indices.length;
174 	
175 	this.renderables.push(renderable);
176 
177 }

181 /**
182  * 	Remove polygon from renderer
183  */
184 GlobWeb.SimplePolygonRenderer.prototype.removeGeometry = function(geometry,style){
185 	
186 	for ( var i = 0; i<this.renderables.length; i++ )
187 	{
188 		var currentRenderable = this.renderables[i];
189 		if ( currentRenderable.geometry == geometry){
190 
191 			// Dispose resources
192 			var gl = this.renderContext.gl;
193 	
194 			if ( currentRenderable.indexBuffer )
195 				gl.deleteBuffer(currentRenderable.indexBuffer);
196 			if ( currentRenderable.vertexBuffer )
197 				gl.deleteBuffer(currentRenderable.vertexBuffer);
198 				
199 			// The texture is managed in a cache, so do not remove it right now
200 			if ( currentRenderable.texture )
201 			{
202 				var cacheTexture = this.textureCache[currentRenderable.textureUrl];
203 				cacheTexture.count--;
204 				if ( cacheTexture.count <= 0 )
205 				{
206 					this.texturesToPurge.push(currentRenderable.textureUrl);
207 				}
208 			}
209 
210 			currentRenderable.indexBuffer = null;
211 			currentRenderable.vertexBuffer = null;
212 			currentRenderable.texture = null;
213 
214 			// Remove from array
215 			this.renderables.splice(i, 1);
216 			break;
217 		}
218 	}
219 }

223 /**
224  * 	Render all the polygons
225  */
226 GlobWeb.SimplePolygonRenderer.prototype.render = function()
227 {
228 	var renderContext = this.renderContext;
229 	var gl = renderContext.gl;
230 	
231 	// Purge textures from the cache if nay
232 	for ( var i = 0; i < this.texturesToPurge.length; i++ )
233 	{
234 		var cacheTexture = this.textureCache[ this.texturesToPurge[i] ];
235 		if ( cacheTexture.count <= 0 )
236 		{
237 			gl.deleteTexture(cacheTexture.texture);
238 			delete this.textureCache[ this.texturesToPurge[i] ];
239 		}
240 	}
241 	this.texturesToPurge.length = 0;
242 
243 	gl.disable(gl.DEPTH_TEST);
244 	gl.enable(gl.BLEND);
245 	gl.blendEquation(gl.FUNC_ADD);
246 	gl.blendFunc(gl.SRC_ALPHA, gl.ONE_MINUS_SRC_ALPHA);
247 	
248 	this.program.apply();
249 
250 	// The shader only needs the viewProjection matrix, use GlobWeb.modelViewMatrix as a temporary storage
251 	mat4.multiply(renderContext.projectionMatrix, renderContext.viewMatrix, renderContext.modelViewMatrix)
252 	gl.uniformMatrix4fv(this.program.uniforms["viewProjectionMatrix"], false, renderContext.modelViewMatrix);
253 	gl.uniform1i(this.program.uniforms["texture"], 0);
254 	gl.activeTexture(gl.TEXTURE0);
255 	
256 	gl.bindBuffer(gl.ARRAY_BUFFER, this.tcoordBuffer);
257 	gl.vertexAttribPointer(this.program.attributes['tcoord'], 2, gl.FLOAT, false, 0, 0);
258 
259 	for ( var n = 0; n < this.renderables.length; n++ )
260 	{
261 		var renderable = this.renderables[n];
262 		
263 		if ( !renderable.layer._visible
264 			|| renderable.layer._opacity <= 0.0 )
265 			continue;
266 			
267 		if ( renderable.textureUrl )
268 		{
269 			gl.uniform4f(this.program.uniforms["u_color"], 1.0, 1.0, 1.0, renderable.layer._opacity);  // use whiteColor
270 		}
271 		else
272 		{
273 			gl.uniform4fv(this.program.uniforms["u_color"], renderable.style.fillColor[0], renderable.style.fillColor[1], renderable.style.fillColor[2], 
274 				renderable.style.fillColor[3] * renderable.layer._opacity);  // use fillColor
275 		}
276 		
277 		if ( renderable.texture ) 
278 			gl.bindTexture(gl.TEXTURE_2D, renderable.texture); // use texture of renderable
279 		else
280 			gl.bindTexture(gl.TEXTURE_2D, this.whiteTexture);  // use white texture
281 				
282 		gl.bindBuffer(gl.ARRAY_BUFFER, renderable.vertexBuffer);
283 		gl.vertexAttribPointer(this.program.attributes['vertex'], renderable.vertexBuffer.itemSize, gl.FLOAT, false, 0, 0);
284 		
285 		gl.bindBuffer(gl.ELEMENT_ARRAY_BUFFER, renderable.indexBuffer);
286 		
287 		gl.drawElements( gl.TRIANGLES, renderable.indexBuffer.numItems, gl.UNSIGNED_SHORT, 0);
288 	}
289 	
290 	gl.enable(gl.DEPTH_TEST);
291 	gl.disable(gl.BLEND);
292 }

298 	creator: function(globe) { 
299 			return new GlobWeb.SimplePolygonRenderer(globe.tileManager);
300 		},
301 	canApply: function(type,style) {return (style.rendererHint == "Basic") && (type == "Polygon") && (style.fill == true); }
