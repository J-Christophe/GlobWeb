 22 /** @constructor
 23 	TileRequest constructor
 24  */
 25 GlobWeb.TileRequest = function(cb)
 26 {
 27 	this.successfull = false;
 28 	this.failed = false;
 29 	this.tile = null;
 30 	this.imageLoaded = false;
 31 	this.elevationLoaded = true;
 32 	this.callback = cb;
 33 
 34 	var that = this;
 35 	
 36 	this.image = new Image();
 37 	this.image.crossOrigin = '';
 38 	this.image.onload = function() { that.handleLoadedImage(); };
 39 	this.image.onerror = function()  { that.handleErrorImage(); };
 40 
 41 	this.xhr = new XMLHttpRequest();
 42 	this.xhr.onreadystatechange = function(e)
 43 	{
 44 		if ( that.xhr.readyState == 4 && that.xhr.status == 200)
 45 		{
 46 			that.handleLoadedElevation( that );
 47 		}
 48 
 49 		if ( that.xhr.status >= 400)
 50 		{
 51 			that.handleErrorElevation( that );
 52 		}
 53 	};
 54 }

 56 /**************************************************************************************************************/

 58 /**
 59 	Handle when image is loaded
 60  */
 61 GlobWeb.TileRequest.prototype.handleLoadedImage = function() 
 62 {
 63 	this.imageLoaded = true;
 64 	if ( this.elevationLoaded )
 65 	{
 66 		this.successfull = true;
 67 		this.callback();
 68 	}
 69 }
 70 
 71 /**************************************************************************************************************/
 72 
 73 /**
 74 	Handle when loading image failed
 75  */
 76 GlobWeb.TileRequest.prototype.handleErrorImage = function() 
 77 {
 78 	console.log( "Error while loading " + this.image.src );
 79 	this.failed = true;
 80 }
 81 
 82 /**************************************************************************************************************/
 83 
 84 /**
 85 	Handle when elevation is loaded
 86  */
 87 GlobWeb.TileRequest.prototype.handleLoadedElevation = function() 
 88 {
 89 	this.elevations = this.xhr.responseText;
 90 		
 91 	this.elevationLoaded = true;
 92 	
 93 	if ( this.imageLoaded )
 94 	{
 95 		this.successfull = true;
 96 		this.callback();
 97 	}
 98 }
 99 
100 /**************************************************************************************************************/
101 
102 /**
103 	Handle when loading elevation failed
104  */
105 GlobWeb.TileRequest.prototype.handleErrorElevation = function() 
106 {
107 	this.elevations = null;
108 	this.elevationLoaded = true;
109 	
110 	if ( this.imageLoaded )
111 	{
112 		this.successfull = true;
113 		this.callback();
114 	}
115 }

117 /**************************************************************************************************************/

119 /**
120 	Launch the HTTP request for a tile
121  */
122 GlobWeb.TileRequest.prototype.launch = function(imageUrl,elevationUrl)
123 {
124 	this.successfull = false;
125 	this.failed = false;
126 	
127 	this.imageLoaded = false;
128 	this.image.src = imageUrl;
129 	
130 	// Request the elevation if needed
131 	if ( elevationUrl )
132 	{
133 		this.elevationLoaded = false;
134 		this.xhr.open("GET", elevationUrl );
135 		this.xhr.send();
136 	}
137 }
