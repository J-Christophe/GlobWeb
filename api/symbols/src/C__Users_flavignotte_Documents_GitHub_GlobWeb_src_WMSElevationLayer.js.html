 23 /** @export
 24 	@constructor
 25 	WMSElevationLayer constructor
 26  */
 27 GlobWeb.WMSElevationLayer = function( options )
 28 {
 29 	options['format'] = 'image/x-aaigrid';
 30 	options['tilePixelSize'] = options['tilePixelSize'] || 33;
 31 	GlobWeb.WMSLayer.prototype.constructor.call( this, options );
 32 }
 33 
 34 GlobWeb.inherits(GlobWeb.WMSLayer,GlobWeb.WMSElevationLayer);
 35

 39 /**
 40 	Parse a elevation response
 41  */
 42 GlobWeb.WMSElevationLayer.prototype.parseElevations = function(text)
 43 {
 44 	var elevations = [];
 45 	var lines = text.trim().split('\n');
 46 	
 47 	for ( var i = 5; i < lines.length; i++ )
 48 	{
 49 		var elts = lines[i].trim().split(/\s+/);
 50 		for ( var n=0; n < elts.length; n++ )
 51 		{
 52 			elevations.push( parseInt(elts[n]) );
 53 		}
 54 	}
 55 	
 56 	return elevations;
 57 }
