 22 /** 
 23 	@constructor
 24  */
 25 GlobWeb.GroundOverlayRenderer = function(tileManager)
 26 {
 27 	this.renderContext = tileManager.renderContext;
 28 	this.tileIndexBuffer = tileManager.tileIndexBuffer;
 29 	
 30 	var vertexShader = "\
 31 	attribute vec3 vertex;\n\
 32 	attribute vec2 tcoord;\n\
 33 	uniform mat4 modelViewMatrix;\n\
 34 	uniform mat4 projectionMatrix;\n\
 35 	uniform vec4 extent; \n\
 36 	\
 37 	varying vec2 texCoord;\n\
 38 	\
 39 	void main(void) \n\
 40 	{\n\
 41 		gl_Position = projectionMatrix * modelViewMatrix * vec4(vertex, 1.0);\n\
 42 		texCoord.x = tcoord.x * (extent[1] - extent[0]) + extent[0];\n\
 43 		texCoord.y = tcoord.y * (extent[3] - extent[2]) + extent[2];\n\
 44 	}\n\
 45 	";
 46 
 47 	var fragmentShader = "\
 48 	#ifdef GL_ES\n\
 49 	precision highp float;\n\
 50 	#endif\n\
 51 	\n\
 52 	varying vec2 texCoord;\n\
 53 	uniform sampler2D overlayTexture;\n\
 54 	uniform mat3 transform;\n\
 55 	uniform float opacity; \n\
 56 	\n\
 57 	void main(void)\n\
 58 	{\n\
 59 		vec3 tc = transform * vec3(texCoord,1.0); \n\
 60 		tc.xy /= tc.z; \n\
 61 		gl_FragColor.rgba = texture2D(overlayTexture, tc.xy); \n\
 62 		gl_FragColor.a = (tc.x >= 0.0 && tc.x <= 1.0 && tc.y >= 0.0 && tc.y <= 1.0) ? opacity * gl_FragColor.a  : 0.0; \n\
 63 	}\n\
 64 	";
 65 	
 66     this.program = new GlobWeb.Program(this.renderContext);
 67 	this.program.createFromSource( vertexShader, fragmentShader );
 68 	
 69 	this.groundOverlays = [];
 70 	
 71 	this.needsOffset = true;
 72 }

 76 /*
 77 	Render the ground overlays above the tiles in parameter
 78  */
 79 GlobWeb.GroundOverlayRenderer.prototype.render = function( tiles )
 80 {
 81  	var gl = this.renderContext.gl;
 82 
 83 	// Setup program
 84     this.program.apply();
 85 	
 86 	var attributes = this.program.attributes;
 87 		
 88 	gl.uniformMatrix4fv(this.program.uniforms["projectionMatrix"], false, this.renderContext.projectionMatrix);
 89 	gl.uniform1i(this.program.uniforms["overlayTexture"], 0);
 90 	gl.enable(gl.BLEND);
 91 	gl.blendFunc( gl.SRC_ALPHA, gl.ONE_MINUS_SRC_ALPHA );
 92 	gl.depthFunc( gl.LEQUAL );
 93 	
 94 	var modelViewMatrix = mat4.create();
 95 	
 96 	var currentIB = null;
 97 
 98 	for ( var j=0; j < this.groundOverlays.length; j++ )
 99 	{
100 		var go = this.groundOverlays[j];
101 		
102 		// Image is not loaded, nothing to be done
103 		if ( !go.image.complete )
104 		{
105 			continue;
106 		}
107 		
108 		if ( !go.texture )
109 		{
110 			if ( go.flipY )
111 				gl.pixelStorei(gl.UNPACK_FLIP_Y_WEBGL, true );
112 			go.texture = this.renderContext.createNonPowerOfTwoTextureFromImage(go.image);
113 			if ( go.flipY )
114 				gl.pixelStorei(gl.UNPACK_FLIP_Y_WEBGL, false );
115 		}
116 		
117 		var initialized = false;
118 		
119 		for ( var i = 0; i < tiles.length; i++ )
120 		{
121 			var tile = tiles[i];
122 			if ( go.geoBound.intersects( tile.geoBound ) )
123 			{
124 				if (!initialized)
125 				{					
126 					gl.uniformMatrix3fv(this.program.uniforms["transform"], false, go.inverseTransform );
127 					gl.uniform1f(this.program.uniforms["opacity"], go.opacity );
128 
129 					gl.activeTexture(gl.TEXTURE0);
130 					gl.bindTexture(gl.TEXTURE_2D, go.texture);
131 					
132 					initialized = true;
133 				}
134 
135 				var extent = (tile.state == GlobWeb.Tile.State.LOADED) ? tile.geoBound : tile.parent.geoBound;
136 				
137 				gl.uniform4f(this.program.uniforms["extent"], extent.west, extent.east, extent.north, extent.south );
138 				mat4.multiply( this.renderContext.viewMatrix, tile.matrix, modelViewMatrix );
139 				gl.uniformMatrix4fv(this.program.uniforms["modelViewMatrix"], false, modelViewMatrix);
140 
141 				// Bind the vertex buffer
142 				gl.bindBuffer(gl.ARRAY_BUFFER, tile.vertexBuffer);
143 				gl.vertexAttribPointer(attributes['vertex'], 3, gl.FLOAT, false, 0, 0);
144 				
145 				// Bind the index buffer only if different (index buffer is shared between tiles)
146 				var indexBuffer = ( tile.state == GlobWeb.Tile.State.LOADED ) ? this.tileIndexBuffer.getSolid() : this.tileIndexBuffer.getSubSolid(tile.parentIndex);
147 				if ( currentIB != indexBuffer )
148 				{
149 					gl.bindBuffer(gl.ELEMENT_ARRAY_BUFFER, indexBuffer);
150 					currentIB = indexBuffer;
151 				}
152 				
153 				// Finally draw the tiles
154 				gl.drawElements(gl.TRIANGLES, indexBuffer.numIndices, gl.UNSIGNED_SHORT, 0);
155 			}
156 		}
157 	}
158 
159 	gl.disable(gl.BLEND);
160 }
